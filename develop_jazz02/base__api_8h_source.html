180 inline int move_const(pChar p_buff, int buff_size, pChar p_url, pChar p_base = nullptr) {
181
182 if (*(p_url++) != '&')
183 return RET_MV_CONST_FAILED;
184
185 int url_len = strlen(p_url) - 1;
186
187 pChar p_end = p_url + url_len;
188
189 int ret = RET_MV_CONST_NOTHING;
190
191 if (*p_end == 'w') {
192 if ((*(--p_end) != 'e') || (*(--p_end) != 'n') || (*(--p_end) != '.') || (*(--p_end) != ';'))
193 return RET_MV_CONST_FAILED;
194
195 ret = RET_MV_CONST_NEW_ENTITY;
196 url_len -= 4;
197 }
198
199 if (*p_end != ';' && *p_end != ']' && *p_end != ')')
200 return RET_MV_CONST_FAILED;
201
202 if (*p_buff != 0)
203 return ret;
204
205 if (p_base != nullptr) {
206 if (buff_size < 4 + SHORT_NAME_SIZE)
207 return RET_MV_CONST_FAILED;
208
209 *(p_buff++) = '/';
210 *(p_buff++) = '/';
211
212 buff_size -= 2;
213
214 for (int i = 0; i < SHORT_NAME_SIZE; i++) {
215 char c = *(p_base++);
216
217 buff_size--;
218
219 if (c == 0) {
220 *(p_buff++) = '/';
221 break;
222 }
223 *(p_buff++) = c;
224 }
225 }
226
227 if (url_len >= buff_size)
228 return RET_MV_CONST_FAILED;
229
230 memcpy(p_buff, p_url, url_len);
231
232 p_buff += url_len;
233
234 *(p_buff) = 0;
235
236 return ret;
237 }

251 inline StatusCode get_right_local(pTransaction &p_txn, ApiQueryState &q_state) {
252
253 pTransaction p_aux;
254 pContainer p_container, p_aux_cont;
255
256 if (q_state.apply == APPLY_ASSIGN_CONST)
257 return block_from_const(p_txn, q_state.url) ? SERVICE_NO_ERROR : SERVICE_ERROR_WRONG_ARGUMENTS;
258
259 p_container = (pContainer) base_server[TenBitsAtAddress(q_state.r_value.base)];
260
261 if (p_container == nullptr)
262 return SERVICE_ERROR_WRONG_BASE;
263
264 switch (q_state.apply) {
265 case APPLY_ASSIGN_NOTHING:
266 return p_container->get(p_txn, q_state.r_value);
267
268 case APPLY_ASSIGN_NAME:
269 return p_container->get(p_txn, q_state.r_value, q_state.name);
270
271 case APPLY_ASSIGN_URL:
272 return p_container->get(p_txn, (pChar) q_state.url);
273
274 case APPLY_ASSIGN_FUNCTION:
275 p_aux_cont = (pContainer) base_server[TenBitsAtAddress(q_state.rr_value.base)];
276
277 if (p_aux_cont == nullptr)
278 return SERVICE_ERROR_WRONG_BASE;
279
280 if (p_aux_cont->get(p_aux, q_state.rr_value) != SERVICE_NO_ERROR)
281 return SERVICE_ERROR_BLOCK_NOT_FOUND;
282
283 if (q_state.r_value.key[0] == 0) {
284 if (p_container->modify(q_state.r_value, (pTuple) p_aux->p_block) != SERVICE_NO_ERROR) {
285 p_aux_cont->destroy_transaction(p_aux);
286
287 return SERVICE_ERROR_IO_ERROR;
288 }
289 Name ent = {"result"};
290 if (new_block(p_txn, (pTuple) p_aux->p_block, ent) != SERVICE_NO_ERROR) {
291 p_aux_cont->destroy_transaction(p_aux);
292
293 return SERVICE_ERROR_IO_ERROR;
294 }
295 } else {
296 if (p_container->exec(p_txn, q_state.r_value, (pTuple) p_aux->p_block) != SERVICE_NO_ERROR) {
297 p_aux_cont->destroy_transaction(p_aux);
298
299 return SERVICE_ERROR_IO_ERROR;
300 }
301 }
302 p_aux_cont->destroy_transaction(p_aux);
303
304 return SERVICE_NO_ERROR;
305
306 case APPLY_ASSIGN_FUNCT_CONST:
307 if (!block_from_const(p_aux, q_state.url, q_state.r_value.key[0] == 0))
308 return SERVICE_ERROR_WRONG_ARGUMENTS;
309
310 if (q_state.r_value.key[0] == 0) {
311 if (p_container->modify(q_state.r_value, (pTuple) p_aux->p_block) != SERVICE_NO_ERROR) {
312 destroy_transaction(p_aux);
313
314 return SERVICE_ERROR_IO_ERROR;
315 }
316 Name ent = {"result"};
317 if (new_block(p_txn, (pTuple) p_aux->p_block, ent) != SERVICE_NO_ERROR) {
318 destroy_transaction(p_aux);
319
320 return SERVICE_ERROR_IO_ERROR;
321 }
322 } else {
323 if (p_container->exec(p_txn, q_state.r_value, (pTuple) p_aux->p_block) != SERVICE_NO_ERROR) {
324 destroy_transaction(p_aux);
325
326 return SERVICE_ERROR_IO_ERROR;
327 }
328 }
329 destroy_transaction(p_aux);
330
331 return SERVICE_NO_ERROR;
332
333 case APPLY_ASSIGN_FILTER:
334 p_aux_cont = (pContainer) base_server[TenBitsAtAddress(q_state.rr_value.base)];
335
336 if (p_aux_cont == nullptr)
337 return SERVICE_ERROR_WRONG_BASE;
338
339 if (p_aux_cont->get(p_aux, q_state.rr_value) != SERVICE_NO_ERROR)
340 return SERVICE_ERROR_BLOCK_NOT_FOUND;
341
342 if (p_container->get(p_txn, q_state.r_value, p_aux->p_block) != SERVICE_NO_ERROR) {
343 p_aux_cont->destroy_transaction(p_aux);
344
345 return SERVICE_ERROR_IO_ERROR;
346 }
347 p_aux_cont->destroy_transaction(p_aux);
348
349 return SERVICE_NO_ERROR;
350
351 case APPLY_ASSIGN_FILT_CONST:
352 if (!block_from_const(p_aux, q_state.url))
353 return SERVICE_ERROR_WRONG_ARGUMENTS;
354
355 if (p_container->get(p_txn, q_state.r_value, p_aux->p_block) != SERVICE_NO_ERROR) {
356 destroy_transaction(p_aux);
357
358 return SERVICE_ERROR_IO_ERROR;
359 }
360 destroy_transaction(p_aux);
361
362 return SERVICE_NO_ERROR;
363
364 case APPLY_ASSIGN_RAW:
365 if (p_container->get(p_aux, q_state.r_value) != SERVICE_NO_ERROR)
366 return SERVICE_ERROR_BLOCK_NOT_FOUND;
367
368 if (new_block(p_txn, p_aux->p_block, CELL_TYPE_UNDEFINED) != SERVICE_NO_ERROR) {
369 p_container->destroy_transaction(p_aux);
370
371 return SERVICE_ERROR_IO_ERROR;
372 }
373 p_container->destroy_transaction(p_aux);
374
375 return SERVICE_NO_ERROR;
376
377 case APPLY_ASSIGN_TEXT:
378 if (p_container->get(p_aux, q_state.r_value) != SERVICE_NO_ERROR)
379 return SERVICE_ERROR_BLOCK_NOT_FOUND;
380
381 if (new_block(p_txn, p_aux->p_block, (pChar) nullptr, true) != SERVICE_NO_ERROR) {
382 p_container->destroy_transaction(p_aux);
383
384 return SERVICE_ERROR_IO_ERROR;
385 }
386 p_container->destroy_transaction(p_aux);
387
388 return SERVICE_NO_ERROR;
389 }
390 return SERVICE_ERROR_MISC_SERVER;
391 }

404 inline StatusCode get_right_remote(pTransaction &p_txn, ApiQueryState &q_state) {
405
406 char buffer_2k[SIZE_BUFFER_REMOTE_CALL];
407
408 switch (q_state.apply) {
409 case APPLY_ASSIGN_NOTHING:
410 sprintf(buffer_2k, "//%s/%s/%s", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key);
411 break;
412 case APPLY_ASSIGN_NAME:
413 sprintf(buffer_2k, "//%s/%s/%s:%s", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key, q_state.name);
414 break;
415 case APPLY_ASSIGN_URL:
416 sprintf(buffer_2k, "//%s&%s;", q_state.r_value.base, q_state.url);
417 break;
418 case APPLY_ASSIGN_FUNCTION:
419 sprintf(buffer_2k, "//%s/%s/%s(//%s/%s/%s)", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key,
420 q_state.rr_value.base, q_state.rr_value.entity, q_state.rr_value.key);
421 break;
422 case APPLY_ASSIGN_FUNCT_CONST:
423 sprintf(buffer_2k, "//%s/%s/%s(&%s)", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key, q_state.url);
424 break;
425 case APPLY_ASSIGN_FILTER:
426 sprintf(buffer_2k, "//%s/%s/%s[//%s/%s/%s]", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key,
427 q_state.rr_value.base, q_state.rr_value.entity, q_state.rr_value.key);
428 break;
429 case APPLY_ASSIGN_FILT_CONST:
430 sprintf(buffer_2k, "//%s/%s/%s[&%s]", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key, q_state.url);
431 break;
432 case APPLY_ASSIGN_RAW:
433 sprintf(buffer_2k, "//%s/%s/%s.raw", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key);
434 break;
435 case APPLY_ASSIGN_TEXT:
436 sprintf(buffer_2k, "//%s/%s/%s.text", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key);
437 break;
438 default:
439 return SERVICE_ERROR_WRONG_ARGUMENTS;
440 }
441 return p_channels->forward_get(p_txn, q_state.r_node, buffer_2k);
442 }
