174 inline int move_const(pChar p_buff, int buff_size, pChar p_url, pChar p_base = nullptr) {
175
176 if (*(p_url++) != '&')
177 return RET_MV_CONST_FAILED;
178
179 int url_len = strlen(p_url) - 1;
180
181 pChar p_end = p_url + url_len;
182
183 int ret = RET_MV_CONST_NOTHING;
184
185 if (*p_end == 'w') {
186 if ((*(--p_end) != 'e') || (*(--p_end) != 'n') || (*(--p_end) != '.') || (*(--p_end) != ';'))
187 return RET_MV_CONST_FAILED;
188
189 ret = RET_MV_CONST_NEW_ENTITY;
190 url_len -= 4;
191 }
192
193 if (*p_end != ';' && *p_end != ']' && *p_end != ')')
194 return RET_MV_CONST_FAILED;
195
196 if (*p_buff != 0)
197 return ret;
198
199 if (p_base != nullptr) {
200 if (buff_size < 4 + SHORT_NAME_SIZE)
201 return RET_MV_CONST_FAILED;
202
203 *(p_buff++) = '/';
204 *(p_buff++) = '/';
205
206 buff_size -= 2;
207
208 for (int i = 0; i < SHORT_NAME_SIZE; i++) {
209 char c = *(p_base++);
210
211 buff_size--;
212
213 if (c == 0) {
214 *(p_buff++) = '/';
215 break;
216 }
217 *(p_buff++) = c;
218 }
219 }
220
221 if (url_len >= buff_size)
222 return RET_MV_CONST_FAILED;
223
224 memcpy(p_buff, p_url, url_len);
225
226 p_buff += url_len;
227
228 *(p_buff) = 0;
229
230 return ret;
231 }
