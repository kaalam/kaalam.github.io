369  inline StatusCode get_right_remote(pTransaction &p_txn, HttpQueryState &q_state) {
370 
371  char buffer_2k[2048];
372 
373  switch (q_state.apply) {
374  case APPLY_ASSIGN_NOTHING:
375  sprintf(buffer_2k, "//%s/%s/%s", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key);
376  break;
377  case APPLY_ASSIGN_NAME:
378  sprintf(buffer_2k, "//%s/%s/%s:%s", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key, q_state.name);
379  break;
380  case APPLY_ASSIGN_URL:
381  sprintf(buffer_2k, "//%s&%s;", q_state.r_value.base, q_state.url);
382  break;
383  case APPLY_ASSIGN_FUNCTION:
384  sprintf(buffer_2k, "//%s/%s/%s(//%s/%s/%s)", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key,
385  q_state.rr_value.base, q_state.rr_value.entity, q_state.rr_value.key);
386  break;
387  case APPLY_ASSIGN_FUNCT_CONST:
388  sprintf(buffer_2k, "//%s/%s/%s(&%s)", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key, q_state.url);
389  break;
390  case APPLY_ASSIGN_FILTER:
391  sprintf(buffer_2k, "//%s/%s/%s[//%s/%s/%s]", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key,
392  q_state.rr_value.base, q_state.rr_value.entity, q_state.rr_value.key);
393  break;
394  case APPLY_ASSIGN_FILT_CONST:
395  sprintf(buffer_2k, "//%s/%s/%s[&%s]", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key, q_state.url);
396  break;
397  case APPLY_ASSIGN_RAW:
398  sprintf(buffer_2k, "//%s/%s/%s.raw", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key);
399  break;
400  case APPLY_ASSIGN_TEXT:
401  sprintf(buffer_2k, "//%s/%s/%s.text", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key);
402  break;
403  default:
404  return SERVICE_ERROR_WRONG_ARGUMENTS;
405  }
406  return p_channels->forward_get(p_txn, q_state.r_node, buffer_2k);
407  }
