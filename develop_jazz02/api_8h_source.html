242 inline StatusCode get_left_local(pTransaction &p_txn, HttpQueryState &q_state) {
243
244 Locator loc;
245 pTransaction p_aux;
246 pContainer p_container, p_aux_cont;
247 Name ent;
248
249 p_container = (pContainer) base_server[TenBitsAtAddress(q_state.base)];
250
251 if (p_container == nullptr)
252 return SERVICE_ERROR_WRONG_BASE;
253
254 switch (q_state.apply) {
255 case APPLY_NOTHING:
256 memcpy(&loc, &q_state.base, SIZE_OF_BASE_ENT_KEY);
257 return p_container->get(p_txn, loc);
258
259 case APPLY_NAME:
260 memcpy(&loc, &q_state.base, SIZE_OF_BASE_ENT_KEY);
261 return p_container->get(p_txn, loc, q_state.name);
262
263 case APPLY_URL:
264 return p_container->get(p_txn, (pChar) q_state.url);
265
266 case APPLY_FUNCTION:
267 p_aux_cont = (pContainer) base_server[TenBitsAtAddress(q_state.r_value.base)];
268
269 if (p_aux_cont == nullptr)
270 return SERVICE_ERROR_WRONG_BASE;
271
272 if (p_aux_cont->get(p_aux, q_state.r_value) != SERVICE_NO_ERROR)
273 return SERVICE_ERROR_BLOCK_NOT_FOUND;
274
275 memcpy(&loc, &q_state.base, SIZE_OF_BASE_ENT_KEY);
276 if (q_state.key[0] == 0) {
277 if (p_container->modify(loc, (pTuple) p_aux->p_block) != SERVICE_NO_ERROR) {
278 p_aux_cont->destroy_transaction(p_aux);
279
280 return SERVICE_ERROR_IO_ERROR;
281 }
282 strcpy(ent, "result");
283 if (new_block(p_txn, (pTuple) p_aux->p_block, ent) != SERVICE_NO_ERROR) {
284 p_aux_cont->destroy_transaction(p_aux);
285
286 return SERVICE_ERROR_IO_ERROR;
287 }
288 } else {
289 if (p_container->exec(p_txn, loc, (pTuple) p_aux->p_block) != SERVICE_NO_ERROR) {
290 p_aux_cont->destroy_transaction(p_aux);
291
292 return SERVICE_ERROR_IO_ERROR;
293 }
294 }
295 p_aux_cont->destroy_transaction(p_aux);
296
297 return SERVICE_NO_ERROR;
298
299 case APPLY_FUNCT_CONST:
300 if (!block_from_const(p_aux, q_state.url, q_state.key[0] == 0))
301 return SERVICE_ERROR_WRONG_ARGUMENTS;
302
303 memcpy(&loc, &q_state.base, SIZE_OF_BASE_ENT_KEY);
304 if (q_state.key[0] == 0) {
305 if (p_container->modify(loc, (pTuple) p_aux->p_block) != SERVICE_NO_ERROR) {
306 destroy_transaction(p_aux);
307
308 return SERVICE_ERROR_IO_ERROR;
309 }
310 strcpy(ent, "result");
311 if (new_block(p_txn, (pTuple) p_aux->p_block, ent) != SERVICE_NO_ERROR) {
312 destroy_transaction(p_aux);
313
314 return SERVICE_ERROR_IO_ERROR;
315 }
316 } else {
317 if (p_container->exec(p_txn, loc, (pTuple) p_aux->p_block) != SERVICE_NO_ERROR) {
318 destroy_transaction(p_aux);
319
320 return SERVICE_ERROR_IO_ERROR;
321 }
322 }
323 destroy_transaction(p_aux);
324
325 return SERVICE_NO_ERROR;
326
327 case APPLY_FILTER:
328 p_aux_cont = (pContainer) base_server[TenBitsAtAddress(q_state.r_value.base)];
329
330 if (p_aux_cont == nullptr)
331 return SERVICE_ERROR_WRONG_BASE;
332
333 if (p_aux_cont->get(p_aux, q_state.r_value) != SERVICE_NO_ERROR)
334 return SERVICE_ERROR_BLOCK_NOT_FOUND;
335
336 memcpy(&loc, &q_state.base, SIZE_OF_BASE_ENT_KEY);
337 if (p_container->get(p_txn, loc, p_aux->p_block) != SERVICE_NO_ERROR) {
338 p_aux_cont->destroy_transaction(p_aux);
339
340 return SERVICE_ERROR_IO_ERROR;
341 }
342 p_aux_cont->destroy_transaction(p_aux);
343
344 return SERVICE_NO_ERROR;
345
346 case APPLY_FILT_CONST:
347 if (!block_from_const(p_aux, q_state.url))
348 return SERVICE_ERROR_WRONG_ARGUMENTS;
349
350 memcpy(&loc, &q_state.base, SIZE_OF_BASE_ENT_KEY);
351 if (p_container->get(p_txn, loc, p_aux->p_block) != SERVICE_NO_ERROR) {
352 destroy_transaction(p_aux);
353
354 return SERVICE_ERROR_IO_ERROR;
355 }
356 destroy_transaction(p_aux);
357
358 return SERVICE_NO_ERROR;
359
360 case APPLY_RAW:
361 memcpy(&loc, &q_state.base, SIZE_OF_BASE_ENT_KEY);
362 if (p_container->get(p_aux, loc) != SERVICE_NO_ERROR)
363 return SERVICE_ERROR_BLOCK_NOT_FOUND;
364
365 if (new_block(p_txn, p_aux->p_block, CELL_TYPE_UNDEFINED) != SERVICE_NO_ERROR) {
366 p_container->destroy_transaction(p_aux);
367
368 return SERVICE_ERROR_IO_ERROR;
369 }
370 p_container->destroy_transaction(p_aux);
371
372 return SERVICE_NO_ERROR;
373
374 case APPLY_TEXT:
375 memcpy(&loc, &q_state.base, SIZE_OF_BASE_ENT_KEY);
376 if (p_container->get(p_aux, loc) != SERVICE_NO_ERROR)
377 return SERVICE_ERROR_BLOCK_NOT_FOUND;
378
379 if (new_block(p_txn, p_aux->p_block) != SERVICE_NO_ERROR) {
380 p_container->destroy_transaction(p_aux);
381
382 return SERVICE_ERROR_IO_ERROR;
383 }
384 p_container->destroy_transaction(p_aux);
385
386 return SERVICE_NO_ERROR;
387 }
388 return SERVICE_ERROR_MISC_SERVER;
389 }

397 inline StatusCode get_right_remote(pTransaction &p_txn, HttpQueryState &q_state) {
398
399 char buffer_2k[2048];
400
401 switch (q_state.apply) {
402 case APPLY_ASSIGN_NOTHING:
403 sprintf(buffer_2k, "//%s/%s/%s", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key);
404 break;
405 case APPLY_ASSIGN_NAME:
406 sprintf(buffer_2k, "//%s/%s/%s:%s", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key, q_state.name);
407 break;
408 case APPLY_ASSIGN_URL:
409 sprintf(buffer_2k, "//%s&%s;", q_state.r_value.base, q_state.url);
410 break;
411 case APPLY_ASSIGN_FUNCTION:
412 sprintf(buffer_2k, "//%s/%s/%s(//%s/%s/%s)", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key,
413 q_state.rr_value.base, q_state.rr_value.entity, q_state.rr_value.key);
414 break;
415 case APPLY_ASSIGN_FUNCT_CONST:
416 sprintf(buffer_2k, "//%s/%s/%s(&%s)", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key, q_state.url);
417 break;
418 case APPLY_ASSIGN_FILTER:
419 sprintf(buffer_2k, "//%s/%s/%s[//%s/%s/%s]", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key,
420 q_state.rr_value.base, q_state.rr_value.entity, q_state.rr_value.key);
421 break;
422 case APPLY_ASSIGN_FILT_CONST:
423 sprintf(buffer_2k, "//%s/%s/%s[&%s]", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key, q_state.url);
424 break;
425 case APPLY_ASSIGN_RAW:
426 sprintf(buffer_2k, "//%s/%s/%s.raw", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key);
427 break;
428 case APPLY_ASSIGN_TEXT:
429 sprintf(buffer_2k, "//%s/%s/%s.text", q_state.r_value.base, q_state.r_value.entity, q_state.r_value.key);
430 break;
431 default:
432 return SERVICE_ERROR_WRONG_ARGUMENTS;
433 }
434 return p_channels->forward_get(p_txn, q_state.r_node, buffer_2k);
435 }

444 inline StatusCode get_right_local(pTransaction &p_txn, HttpQueryState &q_state) {
445
446 pTransaction p_aux;
447 pContainer p_container, p_aux_cont;
448 Name ent;
449

464

467

470

479 p_aux_cont->destroy_transaction(p_aux);

481 return SERVICE_ERROR_IO_ERROR;

491 p_aux_cont->destroy_transaction(p_aux);
492
493 return SERVICE_ERROR_IO_ERROR;
494 }
495 }
496 p_aux_cont->destroy_transaction(p_aux);
497
498 return SERVICE_NO_ERROR;
499
500 case APPLY_ASSIGN_FUNCT_CONST:
501 if (!block_from_const(p_aux, q_state.url, q_state.r_value.key[0] == 0))
502 return SERVICE_ERROR_WRONG_ARGUMENTS;
503
504 if (q_state.r_value.key[0] == 0) {

506 destroy_transaction(p_aux);

518 destroy_transaction(p_aux);
519
520 return SERVICE_ERROR_IO_ERROR;
521 }

523 destroy_transaction(p_aux);
524
525 return SERVICE_NO_ERROR;
526

528 p_aux_cont = (pContainer) base_server[TenBitsAtAddress(q_state.rr_value.base)];
529
530 if (p_aux_cont == nullptr)
531 return SERVICE_ERROR_WRONG_BASE;
532
533 if (p_aux_cont->get(p_aux, q_state.rr_value) != SERVICE_NO_ERROR)
534 return SERVICE_ERROR_BLOCK_NOT_FOUND;
535
536 if (p_container->get(p_txn, q_state.r_value, p_aux->p_block) != SERVICE_NO_ERROR) {
537 p_aux_cont->destroy_transaction(p_aux);
538
539 return SERVICE_ERROR_IO_ERROR;
540 }
541 p_aux_cont->destroy_transaction(p_aux);
542
543 return SERVICE_NO_ERROR;
544
545 case APPLY_ASSIGN_FILT_CONST:
546 if (!block_from_const(p_aux, q_state.url))
547 return SERVICE_ERROR_WRONG_ARGUMENTS;
548
549 if (p_container->get(p_txn, q_state.r_value, p_aux->p_block) != SERVICE_NO_ERROR) {
550 destroy_transaction(p_aux);
551
552 return SERVICE_ERROR_IO_ERROR;
553 }
554 destroy_transaction(p_aux);
555
556 return SERVICE_NO_ERROR;
557
558 case APPLY_ASSIGN_RAW:
559 if (p_container->get(p_aux, q_state.r_value) != SERVICE_NO_ERROR)
560 return SERVICE_ERROR_BLOCK_NOT_FOUND;
561
562 if (new_block(p_txn, p_aux->p_block, CELL_TYPE_UNDEFINED) != SERVICE_NO_ERROR) {
563 p_container->destroy_transaction(p_aux);
564
565 return SERVICE_ERROR_IO_ERROR;
566 }
567 p_container->destroy_transaction(p_aux);
568
569 return SERVICE_NO_ERROR;
570
571 case APPLY_ASSIGN_TEXT:
572 if (p_container->get(p_aux, q_state.r_value) != SERVICE_NO_ERROR)
573 return SERVICE_ERROR_BLOCK_NOT_FOUND;
574

576 p_container->destroy_transaction(p_aux);
577
578 return SERVICE_ERROR_IO_ERROR;
579 }
580 p_container->destroy_transaction(p_aux);
581
582 return SERVICE_NO_ERROR;
583 }
584 return SERVICE_ERROR_MISC_SERVER;
585 }
