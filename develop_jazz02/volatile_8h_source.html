466  inline pVolatileTransaction split(pVolatileTransaction p_item) {
467  // rotate p_prev if there are two p_next children on same level
468 
469  if ( p_item->p_next != nullptr

505  inline pVolatileTransaction insert(pVolatileTransaction p_new, pVolatileTransaction p_tree) {
506  // Do the normal binary tree insertion procedure. Set the result of the
507  // recursive call to the correct child in case a new node was created or the
508  // root of the subtree changes.
509 
510  if (p_tree == nullptr) {
511  p_new->level = 1;
512  p_new->p_prev = nullptr;
513  p_new->p_next = nullptr;
514 
515  return p_new;
516 
517  } else {
518 
519  if (to_left(p_new, p_tree))
520  p_tree->p_prev = insert(p_new, p_tree->p_prev);
521  else
522  p_tree->p_next = insert(p_new, p_tree->p_next);
523  }
524 
525  // Perform skew and then split. The conditionals that determine whether or
526  // not a rotation will occur or not are inside of the procedures, as given above.
527 
528  p_tree = skew(p_tree);
529  p_tree = split(p_tree);
530 
531  return p_tree;
532  };
