343  inline StatusCode put_queue_insert(uint64_t ent_hash, Name &key, double priority, pBlock p_block, int mode) {
344  HashQueueEntMap::iterator it_queue = queue_ent.find(ent_hash);
345 
346  if (it_queue == queue_ent.end())
347  return SERVICE_ERROR_ENTITY_NOT_FOUND;
348 
349  EntityKeyHash ek = {ent_hash, hash(key)};
350  EntKeyVolXctMap::iterator it_item = queue_key.find(ek);
351 
352  if (it_item != queue_key.end()) {
353  if (mode == WRITE_ONLY_IF_NOT_EXISTS)
354  return SERVICE_ERROR_WRITE_FORBIDDEN;
355 
356  pVolatileTransaction p_item = it_item->second;
357 
358  pBlock p_new = block_malloc(p_block->total_bytes);
359  if (p_new == nullptr)
360  return SERVICE_ERROR_NO_MEM;
361 
362  alloc_bytes -= p_item->p_block->total_bytes;
363  free(p_item->p_block);
364 
365  memcpy(p_new, p_block, p_block->total_bytes);
366 
367  p_item->status = BLOCK_STATUS_EMPTY;
368 
369  it_queue->second.p_root = aat_remove(p_item, it_queue->second.p_root);
370 
371  p_item->priority = priority;
372  p_item->times_used++;
373  p_item->p_block = p_new;
374 
375  p_item->p_block->hash64 = 0;
376 
377  p_item->status = BLOCK_STATUS_READY;
378 
379  it_queue->second.p_root = aat_insert(p_item, it_queue->second.p_root);
380 
381  queue_key[ek] = p_item;
382 
383  return SERVICE_NO_ERROR;
384  }
385  if (mode == WRITE_ONLY_IF_EXISTS)
386  return SERVICE_ERROR_WRITE_FORBIDDEN;
387 
388  if (it_queue->second.queue_use == it_queue->second.queue_size) {
389  pVolatileTransaction p_lowest = aat_lowest_priority(it_queue->second.p_root);
390 
391  if (p_lowest->priority >= priority)
392  return SERVICE_ERROR_LOW_PRIORITY;
393 
394  destroy_item(BASE_QUEUE_10BIT, ent_hash, p_lowest);
395  }
396 
397  pTransaction p_txn;
398 
399  int ret = new_transaction(p_txn);
400 
401  if (ret != SERVICE_NO_ERROR)
402  return ret;
403 
404  pBlock p_new = block_malloc(p_block->total_bytes);
405  if (p_new == nullptr) {
406  destroy_transaction(p_txn);
407 
408  return SERVICE_ERROR_NO_MEM;
409  }
410  memcpy(p_new, p_block, p_block->total_bytes);
411 
412  p_new->hash64 = 0;
413 
414  pVolatileTransaction(p_txn)->priority = priority;
415  pVolatileTransaction(p_txn)->times_used = 0;
416  pVolatileTransaction(p_txn)->key_hash = add_name(ek.key_hash, key);
417  pVolatileTransaction(p_txn)->p_block = p_new;
418  pVolatileTransaction(p_txn)->status = BLOCK_STATUS_READY;
419 
420  it_queue->second.p_root = aat_insert((pVolatileTransaction) p_txn, it_queue->second.p_root);
421  it_queue->second.queue_use++;
422 
423  queue_key[ek] = (pVolatileTransaction) p_txn;
424 
425  return SERVICE_NO_ERROR;
426  }

437  inline StatusCode put_in_deque(HashVolXctMap::iterator it_ent, EntityKeyHash &ek, Name &key, pBlock p_block, bool first = false) {
438 
439  pTransaction p_txn;
440  int ret;
441 
442  if ((ret = new_transaction(p_txn)) != SERVICE_NO_ERROR)
443  return ret;
444 
445  pBlock p_new = block_malloc(p_block->total_bytes);
446  if (p_new == nullptr) {
447  destroy_transaction(p_txn);
448 
449  return SERVICE_ERROR_NO_MEM;
450  }
451  memcpy(p_new, p_block, p_block->total_bytes);
452  p_txn->p_block = p_new;
453  p_txn->p_block->hash64 = 0;
454  p_txn->status = BLOCK_STATUS_READY;
455 
456  deque_key[ek] = (pVolatileTransaction) p_txn;
457  pVolatileTransaction(p_txn)->key_hash = add_name(ek.key_hash, key);
458 
459  if (it_ent->second == nullptr) {
460  it_ent->second = (pVolatileTransaction) p_txn;
461  pVolatileTransaction(p_txn)->p_next = (pVolatileTransaction) p_txn;
462  pVolatileTransaction(p_txn)->p_prev = (pVolatileTransaction) p_txn;
463 
464  return SERVICE_NO_ERROR;
465  }
466 
467  if (first) {
468  pVolatileTransaction p_2nd = it_ent->second;
469  pVolatileTransaction p_last = p_2nd->p_prev;
470 
471  pVolatileTransaction(p_txn)->p_next = p_2nd;
472  p_2nd->p_prev = (pVolatileTransaction) p_txn;
473  pVolatileTransaction(p_txn)->p_prev = p_last;
474  p_last->p_next = (pVolatileTransaction) p_txn;
475 
476  it_ent->second = (pVolatileTransaction) p_txn;
477 
478  return SERVICE_NO_ERROR;
479  }
480  pVolatileTransaction p_last = it_ent->second->p_prev;
481 
482  pVolatileTransaction(p_txn)->p_prev = p_last;
483  it_ent->second->p_prev = (pVolatileTransaction) p_txn;
484  pVolatileTransaction(p_txn)->p_next = it_ent->second;
485  p_last->p_next = (pVolatileTransaction) p_txn;
486 
487  return SERVICE_NO_ERROR;
488  }

499  inline StatusCode put_in_tree(HashVolXctMap::iterator it_ent, EntityKeyHash &ek, Name &key, Name &parent, pBlock p_block) {
500 
501  pVolatileTransaction p_root, p_parent = nullptr, p_next = nullptr;;
502 
503  if ((p_root = it_ent->second) != nullptr) {
504  if (tree_key.find(ek) != tree_key.end())
505  return SERVICE_ERROR_WRITE_FORBIDDEN;
506 
507  EntityKeyHash parent_ek = {ek.ent_hash, hash(parent)};
508 
509  EntKeyVolXctMap::iterator it;
510 
511  if ((it = tree_key.find(parent_ek)) == tree_key.end())
512  return SERVICE_ERROR_PARENT_NOT_FOUND;
513 
514  p_parent = it->second;
515  p_next = p_parent->p_child;
516  }
517 
518  pTransaction p_txn;
519  int ret;
520 
521  if ((ret = new_transaction(p_txn)) != SERVICE_NO_ERROR)
522  return ret;
523 
524  pBlock p_new = block_malloc(p_block->total_bytes);
525  if (p_new == nullptr) {
526  destroy_transaction(p_txn);
527 
528  return SERVICE_ERROR_NO_MEM;
529  }
530  memcpy(p_new, p_block, p_block->total_bytes);
531  p_txn->p_block = p_new;
532  p_txn->p_block->hash64 = 0;
533  p_txn->status = BLOCK_STATUS_READY;
534 
535  tree_key[ek] = (pVolatileTransaction) p_txn;
536 
537  pVolatileTransaction(p_txn)->p_parent = p_parent;
538  pVolatileTransaction(p_txn)->p_child = nullptr;
539  pVolatileTransaction(p_txn)->p_next = p_next;
540  pVolatileTransaction(p_txn)->num_wins = 0;
541  pVolatileTransaction(p_txn)->num_visits = 0;
542  pVolatileTransaction(p_txn)->key_hash = add_name(ek.key_hash, key);
543 
544  if (p_root == nullptr)
545  it_ent->second = (pVolatileTransaction) p_txn;
546  else
547  p_parent->p_child = (pVolatileTransaction) p_txn;
548 
549  return SERVICE_NO_ERROR;
550  }

622  inline void destroy_item(int base, uint64_t ent_hash, pVolatileTransaction p_item) {
623 
624  EntityKeyHash ek = {ent_hash, p_item->key_hash};
625  pTransaction p_txn = p_item;
626 
627  switch (base) {
628  case BASE_DEQUE_10BIT: {
629  HashVolXctMap::iterator it_ent = deque_ent.find(ent_hash);
630  if (p_item == it_ent->second) {
631  if (p_item->p_next == p_item)
632  it_ent->second = nullptr;
633  else {
634  p_item->p_next->p_prev = p_item->p_prev;
635  p_item->p_prev->p_next = p_item->p_next;
636  it_ent->second = p_item->p_next;
637  }
638  } else {
639  p_item->p_next->p_prev = p_item->p_prev;
640  p_item->p_prev->p_next = p_item->p_next;
641  }
642  erase_name(ek.key_hash);
643  deque_key.erase(ek);
644  destroy_transaction(p_txn); }
645 
646  return;
647 
648  case BASE_QUEUE_10BIT: {
649  HashQueueEntMap::iterator it_ent = queue_ent.find(ent_hash);
650 
651  it_ent->second.p_root = aat_remove(p_item, it_ent->second.p_root);
652  it_ent->second.queue_use--;
653 
654  erase_name(ek.key_hash);
655  queue_key.erase(ek);
656  destroy_transaction(p_txn); }
657 
658  return;
659 
660  case BASE_TREE_10BIT:
661  erase_name(ek.key_hash);
662  destroy_transaction(p_txn);
663  tree_key.erase(ek);
664  }
665  }

722  inline StatusCode internal_get(pTransaction &p_txn, pString &p_str, uint64_t &pop_ent, Locator &what) {
723 
724  int base;
725  pVolatileTransaction p_root;
726  EntityKeyHash ek;
727  ek.ent_hash = hash(what.entity);
728 
729  pop_ent = 0;
730 
731  switch (base = TenBitsAtAddress(what.base)) {
732  case BASE_DEQUE_10BIT: {
733  HashVolXctMap::iterator it_ent = deque_ent.find(ek.ent_hash);
734 
735  if (it_ent == deque_ent.end())
736  return SERVICE_ERROR_ENTITY_NOT_FOUND;
737 
738  p_root = it_ent->second; }
739 
740  break;
741 
742  case BASE_INDEX_10BIT: {
743  HashVolXctMap::iterator it_ent = index_ent.find(ek.ent_hash);
744 
745  if (it_ent == index_ent.end())
746  return SERVICE_ERROR_ENTITY_NOT_FOUND;
747 
748  p_root = it_ent->second; }
749 
750  break;
751 
752  case BASE_QUEUE_10BIT: {
753  HashQueueEntMap::iterator it_ent = queue_ent.find(ek.ent_hash);
754 
755  if (it_ent == queue_ent.end())
756  return SERVICE_ERROR_ENTITY_NOT_FOUND;
757 
758  p_root = it_ent->second.p_root; }
759 
760  break;
761 
762  case BASE_TREE_10BIT: {
763  HashVolXctMap::iterator it_ent = tree_ent.find(ek.ent_hash);
764 
765  if (it_ent == tree_ent.end())
766  return SERVICE_ERROR_ENTITY_NOT_FOUND;
767 
768  p_root = it_ent->second; }
769 
770  break;
771 
772  default:
773  return SERVICE_ERROR_WRONG_BASE;
774  }
775 
776  if (p_root == nullptr)
777  return SERVICE_ERROR_EMPTY_ENTITY;
778 
779  Name key, second;
780  int command;
781 
782  if (!parse_command(key, command, second, what.key, false))
783  return SERVICE_ERROR_PARSING_COMMAND;
784 
785  switch (command) {
786  case COMMAND_JUST_THE_KEY:
787  switch (base) {
788  case BASE_DEQUE_10BIT: {
789  ek.key_hash = hash(key);
790  EntKeyVolXctMap::iterator it;
791 
792  if ((it = deque_key.find(ek)) == deque_key.end())
793  return SERVICE_ERROR_BLOCK_NOT_FOUND;
794 
795  p_txn = it->second;
796  p_str = nullptr; }
797 
798  return SERVICE_NO_ERROR;
799 
800  case BASE_QUEUE_10BIT: {
801  ek.key_hash = hash(key);
802  EntKeyVolXctMap::iterator it;
803 
804  if ((it = queue_key.find(ek)) == queue_key.end())
805  return SERVICE_ERROR_BLOCK_NOT_FOUND;
806 
807  p_txn = it->second;
808  p_str = nullptr; }
809 
810  return SERVICE_NO_ERROR;
811 
812  case BASE_TREE_10BIT: {
813  ek.key_hash = hash(key);
814  EntKeyVolXctMap::iterator it;
815 
816  if ((it = tree_key.find(ek)) == tree_key.end())
817  return SERVICE_ERROR_BLOCK_NOT_FOUND;
818 
819  p_txn = it->second;
820  p_str = nullptr; }
821 
822  return SERVICE_NO_ERROR;
823 
824  default: {
825  Index::iterator it;
826 
827  if ((it = p_root->p_hea->index.find(key)) == p_root->p_hea->index.end())
828  return SERVICE_ERROR_BLOCK_NOT_FOUND;
829 
830  p_txn = nullptr;
831  p_str = &it->second; }
832 
833  return SERVICE_NO_ERROR;
834  }
835 
836  case COMMAND_CHILD_10BIT: {
837  if (base != BASE_TREE_10BIT)
838  return SERVICE_ERROR_PARSING_COMMAND;
839 
840  ek.key_hash = hash(key);
841  EntKeyVolXctMap::iterator it;
842 
843  if ((it = tree_key.find(ek)) == tree_key.end())
844  return SERVICE_ERROR_BLOCK_NOT_FOUND;
845 
846  p_txn = it->second->p_child;
847 
848  if (p_txn == nullptr)
849  return SERVICE_ERROR_BLOCK_NOT_FOUND;
850 
851  p_str = nullptr; }
852 
853  return SERVICE_NO_ERROR;
854 
855  case COMMAND_PARENT_10BIT: {
856  if (base != BASE_TREE_10BIT)
857  return SERVICE_ERROR_PARSING_COMMAND;
858 
859  ek.key_hash = hash(key);
860  EntKeyVolXctMap::iterator it;
861 
862  if ((it = tree_key.find(ek)) == tree_key.end())
