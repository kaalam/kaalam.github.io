554  inline StatusCode internal_get(pTransaction &p_txn, pString &p_str, uint64_t &pop_ent, Locator &what) {
555 
556  HashVolXctMap *p_ent_map;
557  int base;
558 
559  pop_ent = 0;
560 
561  switch (base = TenBitsAtAddress(what.base)) {
562  case BASE_DEQUE_10BIT:
563  p_ent_map = &deque_ent;
564  break;
565 
566  case BASE_INDEX_10BIT:
567  p_ent_map = &index_ent;
568  break;
569 
570  case BASE_QUEUE_10BIT:
571  p_ent_map = &queue_ent;
572  break;
573 
574  case BASE_TREE_10BIT:
575  p_ent_map = &tree_ent;
576  break;
577 
578  default:
579  return SERVICE_ERROR_WRONG_BASE;
580  }
581 
582  EntityKeyHash ek;
583  ek.ent_hash = hash(what.entity);
584 
585  HashVolXctMap::iterator it_ent = p_ent_map->find(ek.ent_hash);
586 
587  if (it_ent == p_ent_map->end())
588  return SERVICE_ERROR_ENTITY_NOT_FOUND;
589 
590  pVolatileTransaction p_root = it_ent->second;
591 
592  if (p_root == nullptr)
593  return SERVICE_ERROR_EMPTY_ENTITY;
594 
595  Name key, second;
596  int command;
597 
598  if (!parse_command(key, command, second, what.key, false))
599  return SERVICE_ERROR_PARSING_COMMAND;
600 
601  switch (command) {
602  case COMMAND_JUST_THE_KEY:
603  switch (base) {
604  case BASE_DEQUE_10BIT: {
605  ek.key_hash = hash(key);
606  EntKeyVolXctMap::iterator it;
607 
608  if ((it = deque_key.find(ek)) == deque_key.end())
609  return SERVICE_ERROR_BLOCK_NOT_FOUND;
610 
611  p_txn = it->second;
612  p_str = nullptr; }
613 
614  return SERVICE_NO_ERROR;
615 
616  case BASE_QUEUE_10BIT: {
617  ek.key_hash = hash(key);
618  EntKeyVolXctMap::iterator it;
619 
620  if ((it = queue_key.find(ek)) == queue_key.end())
621  return SERVICE_ERROR_BLOCK_NOT_FOUND;
622 
623  p_txn = it->second;
624  p_str = nullptr; }
625 
626  return SERVICE_NO_ERROR;
627 
628  case BASE_TREE_10BIT: {
629  ek.key_hash = hash(key);
630  EntKeyVolXctMap::iterator it;
631 
632  if ((it = tree_key.find(ek)) == tree_key.end())
633  return SERVICE_ERROR_BLOCK_NOT_FOUND;
634 
635  p_txn = it->second;
636  p_str = nullptr; }
637 
638  return SERVICE_NO_ERROR;
639 
640  default: {
641  Index::iterator it;
642 
643  if ((it = p_root->p_hea->index.find(key)) == p_root->p_hea->index.end())
644  return SERVICE_ERROR_BLOCK_NOT_FOUND;
645 
646  p_txn = nullptr;
647  p_str = &it->second; }
648 
649  return SERVICE_NO_ERROR;
650  }
651 
652  case COMMAND_CHILD_10BIT: {
653  if (base != BASE_TREE_10BIT)
654  return SERVICE_ERROR_PARSING_COMMAND;
655 
656  ek.key_hash = hash(key);
657  EntKeyVolXctMap::iterator it;
658 
659  if ((it = tree_key.find(ek)) == tree_key.end())
660  return SERVICE_ERROR_BLOCK_NOT_FOUND;
661 
662  p_txn = it->second->p_child;
663  p_str = nullptr; }
664 
665  return SERVICE_NO_ERROR;
666 
667  case COMMAND_PARENT_10BIT: {
668  if (base != BASE_TREE_10BIT)
669  return SERVICE_ERROR_PARSING_COMMAND;
670 
671  ek.key_hash = hash(key);
672  EntKeyVolXctMap::iterator it;
673 
674  if ((it = tree_key.find(ek)) == tree_key.end())
675  return SERVICE_ERROR_BLOCK_NOT_FOUND;
676 
677  p_txn = it->second->p_parent;
678  p_str = nullptr; }
679 
680  return SERVICE_NO_ERROR;
681 
682  case COMMAND_NEXT_10BIT: {
683  if (base != BASE_TREE_10BIT && base != BASE_DEQUE_10BIT)
684  return SERVICE_ERROR_PARSING_COMMAND;
685 
686  ek.key_hash = hash(key);
687  EntKeyVolXctMap::iterator it;
688 
689  if ((it = tree_key.find(ek)) == tree_key.end())
690  return SERVICE_ERROR_BLOCK_NOT_FOUND;
691 
692  p_txn = it->second->p_next;
693  p_str = nullptr; }
694 
695  return SERVICE_NO_ERROR;
696 
697  case COMMAND_PREV_10BIT: {
698  if (base != BASE_DEQUE_10BIT)
699  return SERVICE_ERROR_PARSING_COMMAND;
700 
701  ek.key_hash = hash(key);
702  EntKeyVolXctMap::iterator it;
703 
704  if ((it = tree_key.find(ek)) == tree_key.end())
705  return SERVICE_ERROR_BLOCK_NOT_FOUND;
706 
707  p_txn = it->second->p_prev;
708  p_str = nullptr; }
709 
710  return SERVICE_NO_ERROR;
711 
712  case COMMAND_HIGH_10BIT:
713  case COMMAND_XHIGH_10BIT: {
714  if (base != BASE_QUEUE_10BIT)
715  return SERVICE_ERROR_PARSING_COMMAND;
716 
717  if (p_root == nullptr)
718  return SERVICE_ERROR_EMPTY_ENTITY;
719 
720  p_txn = highest_priority(p_root);
721  p_str = nullptr;
722 
723  if (command == COMMAND_XHIGH_10BIT)
724  pop_ent = ek.ent_hash; }
725 
726  return SERVICE_NO_ERROR;
727 
728  case COMMAND_LOW_10BIT:
729  case COMMAND_XLOW_10BIT: {
730  if (base != BASE_QUEUE_10BIT)
731  return SERVICE_ERROR_PARSING_COMMAND;
732 
733  if (p_root == nullptr)
734  return SERVICE_ERROR_EMPTY_ENTITY;
735 
736  p_txn = lowest_priority(p_root);
737  p_str = nullptr;
738 
739  if (command == COMMAND_XLOW_10BIT)
740  pop_ent = ek.ent_hash; }
741 
742  return SERVICE_NO_ERROR;
743 
744  case COMMAND_FIRST_10BIT:
745  case COMMAND_PFIRST_10BIT: {
746  if (base != BASE_DEQUE_10BIT && (base != BASE_TREE_10BIT || command == COMMAND_PFIRST_10BIT))
747  return SERVICE_ERROR_PARSING_COMMAND;
748 
749  if (p_root == nullptr)
750  return SERVICE_ERROR_EMPTY_ENTITY;
751 
752  p_txn = p_root;
753  p_str = nullptr;
754 
755  if (command == COMMAND_PFIRST_10BIT)
756  pop_ent = ek.ent_hash; }
757 
758  return SERVICE_NO_ERROR;
759 
760  case COMMAND_LAST_10BIT:
761  case COMMAND_PLAST_10BIT: {
762  if (base != BASE_DEQUE_10BIT)
763  return SERVICE_ERROR_PARSING_COMMAND;
764 
765  if (p_root == nullptr)
766  return SERVICE_ERROR_EMPTY_ENTITY;
767 
768  p_txn = p_root->p_prev;
769  p_str = nullptr;
770 
771  if (command == COMMAND_PLAST_10BIT)
772  pop_ent = ek.ent_hash; }
773 
774  return SERVICE_NO_ERROR;
775 
776  case COMMAND_GET_10BIT:
777  if (base != BASE_INDEX_10BIT)
778  return SERVICE_ERROR_PARSING_COMMAND;
779 
780  p_txn = nullptr;
781  p_str = nullptr;
782  pop_ent = ek.ent_hash;
783 
784  return SERVICE_NO_ERROR;
785  }
786 
787  return SERVICE_ERROR_PARSING_COMMAND;
788  }

804  inline bool parse_command(Name &key_out, int &command, Name &second, Name key_in, bool is_put) {
805  if (key_in[0] == '~') {
806  key_out[0] = 0;
807  second [0] = 0;
808 
809  switch (command = TenBitsAtAddress(&key_in[1])) {
810  case COMMAND_FIRST_10BIT:
811  case COMMAND_LAST_10BIT:
812  return true;
813 
814  case COMMAND_PUT_10BIT:
815  return is_put;
816 
817  case COMMAND_PFIRST_10BIT:
818  case COMMAND_PLAST_10BIT:
819  case COMMAND_HIGH_10BIT:
820  case COMMAND_LOW_10BIT:
821  case COMMAND_GET_10BIT:
822  case COMMAND_XHIGH_10BIT:
823  case COMMAND_XLOW_10BIT:
824  return !is_put;
825 
826  default:
827  if (!is_put || key_in[1] < '0' || key_in[1] > '9')
828  return false;
829 
830  int size, r_len;
831 
832  if (sscanf(&key_in[1], "%d%n", &size, &r_len) != 1 || size <= 0 || key_in[r_len + 1] != 0)
833  return false;
834 
835  command = COMMAND_SIZE + size;
836 
837  return true;
838  }
839  }
840  strcpy(key_out, key_in);
841  pChar pc = strchr(key_out, '~');
842 
843  if (pc == nullptr) {
844  command = COMMAND_JUST_THE_KEY;
845  second[0] = 0;
846 
847  return true;
848  }
849  (*pc++) = 0;
850 
851  if (*pc == 0)
852  return false;
853 
854  if (is_put) {
855  if (!valid_name(pc))
856  return false;
857 
858  strcpy(second, pc);
859  command = COMMAND_SECOND_ARG;
860 
861  return true;
862  }
863  second[0] = 0;
864 
865  switch (command = TenBitsAtAddress(pc)) {
866  case COMMAND_CHILD_10BIT:
867  case COMMAND_NEXT_10BIT:
868  case COMMAND_PARENT_10BIT:
869  case COMMAND_PREV_10BIT:
870  return true;
871  }
872 
873  return false;
874  }
