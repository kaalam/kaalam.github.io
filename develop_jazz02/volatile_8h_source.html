112struct EntityKeyHash {
113 uint64_t ent_hash, key_hash;

115 bool operator==(const EntityKeyHash &o) const {
116 return ent_hash == o.ent_hash && key_hash == o.key_hash;

118
119 bool operator<(const EntityKeyHash &o) const {
120 return ent_hash < o.ent_hash || (ent_hash == o.ent_hash && key_hash < o.key_hash);

314 inline StatusCode populate_index(Index &index, pBlock p_block) {
315
316 if (p_block->cell_type != CELL_TYPE_TUPLE_ITEM || p_block->size != 2)
317 return SERVICE_ERROR_BAD_BLOCK;

342 inline StatusCode put_queue_insert(uint64_t ent_hash, Name &key, double priority, pBlock p_block, int mode) {
343 HashQueueEntMap::iterator it_queue = queue_ent.find(ent_hash);
344
345 if (it_queue == queue_ent.end())
346 return SERVICE_ERROR_ENTITY_NOT_FOUND;
347
348 EntityKeyHash ek = {ent_hash, hash(key)};
349 EntKeyVolXctMap::iterator it_item = queue_key.find(ek);
350
351 if (it_item != queue_key.end()) {
352 if (mode & WRITE_ONLY_IF_NOT_EXISTS)
353 return SERVICE_ERROR_WRITE_FORBIDDEN;
354
355 pVolatileTransaction p_item = it_item->second;
356
357 pBlock p_new = block_malloc(p_block->total_bytes);
358 if (p_new == nullptr)
359 return SERVICE_ERROR_NO_MEM;
360
361 alloc_bytes -= p_item->p_block->total_bytes;
362 free(p_item->p_block);
363
364 memcpy(p_new, p_block, p_block->total_bytes);
365
366 p_item->status = BLOCK_STATUS_EMPTY;
367
368 it_queue->second.p_root = aat_remove(p_item, it_queue->second.p_root);
369
370 p_item->priority = priority;
371 p_item->times_used++;
372 p_item->p_block = p_new;
373
374 p_item->p_block->hash64 = 0;
375
376 p_item->status = BLOCK_STATUS_READY;
377
378 it_queue->second.p_root = aat_insert(p_item, it_queue->second.p_root);
379
380 queue_key[ek] = p_item;
381
382 return SERVICE_NO_ERROR;
383 }
384 if (mode & WRITE_ONLY_IF_EXISTS)
385 return SERVICE_ERROR_WRITE_FORBIDDEN;
386
387 if (it_queue->second.queue_use == it_queue->second.queue_size) {
388 pVolatileTransaction p_lowest = aat_lowest_priority(it_queue->second.p_root);
389
390 if (p_lowest->priority >= priority)
391 return SERVICE_ERROR_LOW_PRIORITY;
392
393 destroy_item(BASE_QUEUE_10BIT, ent_hash, p_lowest);
394 }
395
396 pTransaction p_txn;
397
398 int ret = new_transaction(p_txn);
399
400 if (ret != SERVICE_NO_ERROR)
401 return ret;
402
403 pBlock p_new = block_malloc(p_block->total_bytes);
404 if (p_new == nullptr) {
405 destroy_transaction(p_txn);
406
407 return SERVICE_ERROR_NO_MEM;
408 }
409 memcpy(p_new, p_block, p_block->total_bytes);
410
411 p_new->hash64 = 0;
412
413 pVolatileTransaction(p_txn)->priority = priority;
414 pVolatileTransaction(p_txn)->times_used = 0;
415 pVolatileTransaction(p_txn)->key_hash = add_name(ek.key_hash, key);
416 pVolatileTransaction(p_txn)->p_block = p_new;
417 pVolatileTransaction(p_txn)->status = BLOCK_STATUS_READY;
418
419 it_queue->second.p_root = aat_insert((pVolatileTransaction) p_txn, it_queue->second.p_root);
420 it_queue->second.queue_use++;
421
422 queue_key[ek] = (pVolatileTransaction) p_txn;
423
424 return SERVICE_NO_ERROR;
425 }

436 inline StatusCode put_in_deque(HashVolXctMap::iterator it_ent, EntityKeyHash &ek, Name &key, pBlock p_block, bool first = false) {
437
438 pTransaction p_txn;
439 int ret;
440
441 if ((ret = new_transaction(p_txn)) != SERVICE_NO_ERROR)
442 return ret;
443
444 pBlock p_new = block_malloc(p_block->total_bytes);
445 if (p_new == nullptr) {
446 destroy_transaction(p_txn);
447
448 return SERVICE_ERROR_NO_MEM;
449 }
450 memcpy(p_new, p_block, p_block->total_bytes);
451 p_txn->p_block = p_new;
452 p_txn->p_block->hash64 = 0;
453 p_txn->status = BLOCK_STATUS_READY;
454
455 deque_key[ek] = (pVolatileTransaction) p_txn;
456 pVolatileTransaction(p_txn)->key_hash = add_name(ek.key_hash, key);
457
458 if (it_ent->second == nullptr) {
459 it_ent->second = (pVolatileTransaction) p_txn;
460 pVolatileTransaction(p_txn)->p_next = (pVolatileTransaction) p_txn;
461 pVolatileTransaction(p_txn)->p_prev = (pVolatileTransaction) p_txn;
462
463 return SERVICE_NO_ERROR;
464 }
465
466 if (first) {
467 pVolatileTransaction p_2nd = it_ent->second;
468 pVolatileTransaction p_last = p_2nd->p_prev;
469
470 pVolatileTransaction(p_txn)->p_next = p_2nd;
471 p_2nd->p_prev = (pVolatileTransaction) p_txn;
472 pVolatileTransaction(p_txn)->p_prev = p_last;
473 p_last->p_next = (pVolatileTransaction) p_txn;
474
475 it_ent->second = (pVolatileTransaction) p_txn;
476
477 return SERVICE_NO_ERROR;
478 }
479 pVolatileTransaction p_last = it_ent->second->p_prev;
480
481 pVolatileTransaction(p_txn)->p_prev = p_last;
482 it_ent->second->p_prev = (pVolatileTransaction) p_txn;
483 pVolatileTransaction(p_txn)->p_next = it_ent->second;
484 p_last->p_next = (pVolatileTransaction) p_txn;
485
486 return SERVICE_NO_ERROR;
487 }

498 inline StatusCode put_in_tree(HashVolXctMap::iterator it_ent, EntityKeyHash &ek, Name &key, Name &parent, pBlock p_block) {
499
500 pVolatileTransaction p_root, p_parent = nullptr, p_next = nullptr;;
501
502 if ((p_root = it_ent->second) != nullptr) {
503 if (tree_key.find(ek) != tree_key.end())
504 return SERVICE_ERROR_WRITE_FORBIDDEN;
505
506 EntityKeyHash parent_ek = {ek.ent_hash, hash(parent)};
507
508 EntKeyVolXctMap::iterator it;
509
510 if ((it = tree_key.find(parent_ek)) == tree_key.end())
511 return SERVICE_ERROR_PARENT_NOT_FOUND;
512
513 p_parent = it->second;
514 p_next = p_parent->p_child;
515 }
516
517 pTransaction p_txn;
518 int ret;
519
520 if ((ret = new_transaction(p_txn)) != SERVICE_NO_ERROR)
521 return ret;
522
523 pBlock p_new = block_malloc(p_block->total_bytes);
524 if (p_new == nullptr) {
525 destroy_transaction(p_txn);
526
527 return SERVICE_ERROR_NO_MEM;
528 }
529 memcpy(p_new, p_block, p_block->total_bytes);
530 p_txn->p_block = p_new;
531 p_txn->p_block->hash64 = 0;
532 p_txn->status = BLOCK_STATUS_READY;
533
534 tree_key[ek] = (pVolatileTransaction) p_txn;
535
536 pVolatileTransaction(p_txn)->p_parent = p_parent;
537 pVolatileTransaction(p_txn)->p_child = nullptr;
538 pVolatileTransaction(p_txn)->p_next = p_next;
539 pVolatileTransaction(p_txn)->num_wins = 0;
540 pVolatileTransaction(p_txn)->num_visits = 0;
541 pVolatileTransaction(p_txn)->key_hash = add_name(ek.key_hash, key);
542
543 if (p_root == nullptr)
544 it_ent->second = (pVolatileTransaction) p_txn;
545 else
546 p_parent->p_child = (pVolatileTransaction) p_txn;
547
548 return SERVICE_NO_ERROR;
549 }

621 inline void destroy_item(int base, uint64_t ent_hash, pVolatileTransaction p_item) {
622
623 EntityKeyHash ek = {ent_hash, p_item->key_hash};
624 pTransaction p_txn = p_item;
625
626 switch (base) {
627 case BASE_DEQUE_10BIT: {
628 HashVolXctMap::iterator it_ent = deque_ent.find(ent_hash);
629 if (p_item == it_ent->second) {
630 if (p_item->p_next == p_item)
631 it_ent->second = nullptr;
632 else {
633 p_item->p_next->p_prev = p_item->p_prev;
634 p_item->p_prev->p_next = p_item->p_next;
635 it_ent->second = p_item->p_next;
636 }
637 } else {
638 p_item->p_next->p_prev = p_item->p_prev;
639 p_item->p_prev->p_next = p_item->p_next;
640 }
641 erase_name(ek.key_hash);
642 deque_key.erase(ek);
643 destroy_transaction(p_txn); }
644
645 return;
646
647 case BASE_QUEUE_10BIT: {
648 HashQueueEntMap::iterator it_ent = queue_ent.find(ent_hash);
649
650 it_ent->second.p_root = aat_remove(p_item, it_ent->second.p_root);
651 it_ent->second.queue_use--;
652
653 erase_name(ek.key_hash);
654 queue_key.erase(ek);
655 destroy_transaction(p_txn); }
656
657 return;
658
659 case BASE_TREE_10BIT:
660 erase_name(ek.key_hash);
661 destroy_transaction(p_txn);
662 tree_key.erase(ek);
663 }
664 }

674 inline uint64_t add_name(uint64_t hash, Name &key) {
675
676 HashNameUseMap::iterator it = name.find(hash);
677
678 if (it != name.end())
679 name[hash].use++;
680 else {
681 NameUse nu;
682
683 nu.use = 1;
684 memcpy(&nu.name, &key, sizeof(Name));
685
686 name[hash] = nu;
687 }
688 return hash;
689 }

697 inline void erase_name(uint64_t hash) {
698
699 HashNameUseMap::iterator it = name.find(hash);
700
701 if (it == name.end())
702 return;
703
704 if (--name[hash].use == 0)
705 name.erase(it);
706 }

721 inline StatusCode internal_get(pTransaction &p_txn, pString &p_str, uint64_t &pop_ent, Locator &what) {
722
723 int base;
724 pVolatileTransaction p_root;
725 EntityKeyHash ek;
726 ek.ent_hash = hash(what.entity);
727
728 pop_ent = 0;
729
730 switch (base = TenBitsAtAddress(what.base)) {
731 case BASE_DEQUE_10BIT: {
732 HashVolXctMap::iterator it_ent = deque_ent.find(ek.ent_hash);
733
734 if (it_ent == deque_ent.end())
735 return SERVICE_ERROR_ENTITY_NOT_FOUND;
736
737 p_root = it_ent->second; }
738
739 break;
740
741 case BASE_INDEX_10BIT: {
742 HashVolXctMap::iterator it_ent = index_ent.find(ek.ent_hash);
743
744 if (it_ent == index_ent.end())
745 return SERVICE_ERROR_ENTITY_NOT_FOUND;
746
747 p_root = it_ent->second; }
748
749 break;
750
751 case BASE_QUEUE_10BIT: {
752 HashQueueEntMap::iterator it_ent = queue_ent.find(ek.ent_hash);
753
754 if (it_ent == queue_ent.end())
755 return SERVICE_ERROR_ENTITY_NOT_FOUND;
756
757 p_root = it_ent->second.p_root; }
758
759 break;
760
761 case BASE_TREE_10BIT: {
762 HashVolXctMap::iterator it_ent = tree_ent.find(ek.ent_hash);
763
764 if (it_ent == tree_ent.end())
765 return SERVICE_ERROR_ENTITY_NOT_FOUND;
766
767 p_root = it_ent->second; }
768
769 break;
770
771 default:
772 return SERVICE_ERROR_WRONG_BASE;
773 }
774
775 if (p_root == nullptr)
776 return SERVICE_ERROR_EMPTY_ENTITY;
777
778 Name key, second;
779 int command;
780
781 if (!parse_command(key, command, second, what.key, false))
782 return SERVICE_ERROR_PARSING_COMMAND;
783
784 switch (command) {
785 case COMMAND_JUST_THE_KEY:
786 switch (base) {
787 case BASE_DEQUE_10BIT: {
788 ek.key_hash = hash(key);
789 EntKeyVolXctMap::iterator it;
790
791 if ((it = deque_key.find(ek)) == deque_key.end())
792 return SERVICE_ERROR_BLOCK_NOT_FOUND;
793
794 p_txn = it->second;
795 p_str = nullptr; }
796
797 return SERVICE_NO_ERROR;
798
799 case BASE_QUEUE_10BIT: {
800 ek.key_hash = hash(key);
801 EntKeyVolXctMap::iterator it;
802
803 if ((it = queue_key.find(ek)) == queue_key.end())
804 return SERVICE_ERROR_BLOCK_NOT_FOUND;
805
806 p_txn = it->second;
807 p_str = nullptr; }
808
809 return SERVICE_NO_ERROR;
810
811 case BASE_TREE_10BIT: {
812 ek.key_hash = hash(key);
813 EntKeyVolXctMap::iterator it;
814
815 if ((it = tree_key.find(ek)) == tree_key.end())
816 return SERVICE_ERROR_BLOCK_NOT_FOUND;
817
818 p_txn = it->second;
819 p_str = nullptr; }
820
821 return SERVICE_NO_ERROR;
822
823 default: {
824 Index::iterator it;
825
826 if ((it = p_root->p_hea->index.find(key)) == p_root->p_hea->index.end())
827 return SERVICE_ERROR_BLOCK_NOT_FOUND;
828
829 p_txn = nullptr;
830 p_str = &it->second; }
831
832 return SERVICE_NO_ERROR;
833 }
834
835 case COMMAND_CHILD_10BIT: {
836 if (base != BASE_TREE_10BIT)
837 return SERVICE_ERROR_PARSING_COMMAND;
838
839 ek.key_hash = hash(key);
840 EntKeyVolXctMap::iterator it;
841
842 if ((it = tree_key.find(ek)) == tree_key.end())
843 return SERVICE_ERROR_BLOCK_NOT_FOUND;
844
845 p_txn = it->second->p_child;
846
847 if (p_txn == nullptr)
848 return SERVICE_ERROR_BLOCK_NOT_FOUND;
849
850 p_str = nullptr; }
851
852 return SERVICE_NO_ERROR;
853
854 case COMMAND_PARENT_10BIT: {
855 if (base != BASE_TREE_10BIT)
856 return SERVICE_ERROR_PARSING_COMMAND;
857
858 ek.key_hash = hash(key);
859 EntKeyVolXctMap::iterator it;
860
861 if ((it = tree_key.find(ek)) == tree_key.end())
862 return SERVICE_ERROR_BLOCK_NOT_FOUND;
863
864 p_txn = it->second->p_parent;
865
866 if (p_txn == nullptr)
867 return SERVICE_ERROR_BLOCK_NOT_FOUND;
868
869 p_str = nullptr; }
870
871 return SERVICE_NO_ERROR;
872
873 case COMMAND_NEXT_10BIT: {
874 ek.key_hash = hash(key);
875 EntKeyVolXctMap::iterator it;
876
877 switch (base) {
878 case BASE_TREE_10BIT: {
879 if ((it = tree_key.find(ek)) == tree_key.end())
880 return SERVICE_ERROR_BLOCK_NOT_FOUND;
881
882 p_txn = it->second->p_next;
883
884 if (p_txn == nullptr)
885 return SERVICE_ERROR_BLOCK_NOT_FOUND;
886
887 p_str = nullptr; }
888
889 return SERVICE_NO_ERROR;
890
891 case BASE_DEQUE_10BIT: {
892 if ((it = deque_key.find(ek)) == deque_key.end())
893 return SERVICE_ERROR_BLOCK_NOT_FOUND;
894
895 p_txn = it->second->p_next;
896 p_str = nullptr; }
897
898 return SERVICE_NO_ERROR;
899 }
900 return SERVICE_ERROR_PARSING_COMMAND; }
901
902 case COMMAND_PREV_10BIT: {
903 if (base != BASE_DEQUE_10BIT)
904 return SERVICE_ERROR_PARSING_COMMAND;
905
906 ek.key_hash = hash(key);
907 EntKeyVolXctMap::iterator it;
908

1009 inline bool parse_command(Name &key_out, int &command, Name &second, Name key_in, bool is_put) {
1010 if (key_in[0] == '~') {
1011 key_out[0] = 0;
1012 second [0] = 0;
1013
1014 switch (command = TenBitsAtAddress(&key_in[1])) {
1015 case COMMAND_FIRST_10BIT:
1016 case COMMAND_LAST_10BIT:
1017 return true;
1018
1019 case COMMAND_PUT_10BIT:
1020 return is_put;
1021
1022 case COMMAND_PFIRST_10BIT:
1023 case COMMAND_PLAST_10BIT:
1024 case COMMAND_HIGH_10BIT:
1025 case COMMAND_LOW_10BIT:
1026 case COMMAND_GET_10BIT:
1027 case COMMAND_XHIGH_10BIT:
1028 case COMMAND_XLOW_10BIT:
1029 return !is_put;
1030
1031 default:
1032 if (!is_put || key_in[1] < '0' || key_in[1] > '9')
1033 return false;
1034
1035 int size, r_len;
1036
1037 if (sscanf(&key_in[1], "%d%n", &size, &r_len) != 1 || size <= 0 || key_in[r_len + 1] != 0)
1038 return false;
1039
1040 command = COMMAND_SIZE + size;
1041
1042 return true;
1043 }
1044 }
1045 strcpy(key_out, key_in);
1046 pChar pc = strchr(key_out, '~');
1047
1048 if (pc == nullptr) {
1049 command = COMMAND_JUST_THE_KEY;
1050 second[0] = 0;
1051
1052 return true;
1053 }
1054 (*pc++) = 0;
1055
1056 if (*pc == 0)
1057 return false;
1058
1059 if (is_put) {
1060 strcpy(second, pc);
1061 command = COMMAND_SECOND_ARG;
1062
1063 return true;
1064 }
1065 second[0] = 0;
1066
1067 switch (command = TenBitsAtAddress(pc)) {
1068 case COMMAND_CHILD_10BIT:
1069 case COMMAND_NEXT_10BIT:
1070 case COMMAND_PARENT_10BIT:
1071 case COMMAND_PREV_10BIT:
1072 return true;
1073 }
1074
1075 return false;
1076 }
