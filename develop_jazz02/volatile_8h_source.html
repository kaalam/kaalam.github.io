327 inline StatusCode populate_index(Index &index, pBlock p_block) {
328
329 if (p_block->cell_type != CELL_TYPE_TUPLE || p_block->size != 2)
330 return SERVICE_ERROR_BAD_BLOCK;
331
332 pBlock p_key = pTuple(p_block)->get_block(0);
333 pBlock p_val = pTuple(p_block)->get_block(1);
334
335 if ( p_key->cell_type != CELL_TYPE_STRING || p_val->cell_type != CELL_TYPE_STRING
336 || p_key->rank != 1 || p_val->rank != 1 || p_key->size != p_val->size)
337 return SERVICE_ERROR_BAD_BLOCK;
338
339 for (int i = 0; i < p_key->size; i++)
340 index[p_key->get_string(i)] = p_val->get_string(i);
341
342 return SERVICE_NO_ERROR;
343 }

357 inline StatusCode put_queue_insert(uint64_t ent_hash, Name &key, double priority, pBlock p_block, int mode) {
358 HashQueueEntMap::iterator it_queue = queue_ent.find(ent_hash);
359
360 if (it_queue == queue_ent.end())
361 return SERVICE_ERROR_ENTITY_NOT_FOUND;
362
363 EntityKeyHash ek = {ent_hash, hash(key)};
364 EntKeyVolXctMap::iterator it_item = queue_key.find(ek);
365
366 if (it_item != queue_key.end()) {
367 if (mode & WRITE_ONLY_IF_NOT_EXISTS)
368 return SERVICE_ERROR_WRITE_FORBIDDEN;
369
370 pVolatileTransaction p_item = it_item->second;
371
372 pBlock p_new = block_malloc(p_block->total_bytes);
373 if (p_new == nullptr)
374 return SERVICE_ERROR_NO_MEM;
375
376 alloc_bytes -= p_item->p_block->total_bytes;
377 free(p_item->p_block);
378
379 memcpy(p_new, p_block, p_block->total_bytes);
380
381 p_item->status = BLOCK_STATUS_EMPTY;
382
383 it_queue->second.p_root = aat_remove(p_item, it_queue->second.p_root);
384
385 p_item->priority = priority;
386 p_item->times_used++;
387 p_item->p_block = p_new;
388
389 p_item->p_block->hash64 = 0;
390
391 p_item->status = BLOCK_STATUS_READY;
392
393 it_queue->second.p_root = aat_insert(p_item, it_queue->second.p_root);
394
395 queue_key[ek] = p_item;
396
397 return SERVICE_NO_ERROR;
398 }
399 if (mode & WRITE_ONLY_IF_EXISTS)
400 return SERVICE_ERROR_WRITE_FORBIDDEN;
401
402 if (it_queue->second.queue_use == it_queue->second.queue_size) {
403 pVolatileTransaction p_lowest = aat_lowest_priority(it_queue->second.p_root);
404
405 if (p_lowest->priority >= priority)
406 return SERVICE_ERROR_LOW_PRIORITY;
407
408 destroy_item(BASE_QUEUE_10BIT, ent_hash, p_lowest);
409 }
410
411 pTransaction p_txn;
412
413 int ret = new_transaction(p_txn);
414
415 if (ret != SERVICE_NO_ERROR)
416 return ret;
417
418 pBlock p_new = block_malloc(p_block->total_bytes);
419 if (p_new == nullptr) {
420 destroy_transaction(p_txn);
421
422 return SERVICE_ERROR_NO_MEM;
423 }
424 memcpy(p_new, p_block, p_block->total_bytes);
425
426 p_new->hash64 = 0;
427
428 pVolatileTransaction(p_txn)->priority = priority;
429 pVolatileTransaction(p_txn)->times_used = 0;
430 pVolatileTransaction(p_txn)->key_hash = add_name(ek.key_hash, key);
431 pVolatileTransaction(p_txn)->p_block = p_new;
432 pVolatileTransaction(p_txn)->status = BLOCK_STATUS_READY;
433
434 it_queue->second.p_root = aat_insert((pVolatileTransaction) p_txn, it_queue->second.p_root);
435 it_queue->second.queue_use++;
436
437 queue_key[ek] = (pVolatileTransaction) p_txn;
438
439 return SERVICE_NO_ERROR;
440 }

453 inline StatusCode put_in_deque(HashVolXctMap::iterator it_ent, EntityKeyHash &ek, Name &key, pBlock p_block, bool first = false) {
454
455 pTransaction p_txn;
456 int ret;
457
458 if ((ret = new_transaction(p_txn)) != SERVICE_NO_ERROR)
459 return ret;
460
461 pBlock p_new = block_malloc(p_block->total_bytes);
462 if (p_new == nullptr) {
463 destroy_transaction(p_txn);
464
465 return SERVICE_ERROR_NO_MEM;
466 }
467 memcpy(p_new, p_block, p_block->total_bytes);
468 p_txn->p_block = p_new;
469 p_txn->p_block->hash64 = 0;
470 p_txn->status = BLOCK_STATUS_READY;
471
472 deque_key[ek] = (pVolatileTransaction) p_txn;
473 pVolatileTransaction(p_txn)->key_hash = add_name(ek.key_hash, key);
474
475 if (it_ent->second == nullptr) {
476 it_ent->second = (pVolatileTransaction) p_txn;
477 pVolatileTransaction(p_txn)->p_next = (pVolatileTransaction) p_txn;
478 pVolatileTransaction(p_txn)->p_prev = (pVolatileTransaction) p_txn;
479
480 return SERVICE_NO_ERROR;
481 }
482
483 if (first) {
484 pVolatileTransaction p_2nd = it_ent->second;
485 pVolatileTransaction p_last = p_2nd->p_prev;
486
487 pVolatileTransaction(p_txn)->p_next = p_2nd;
488 p_2nd->p_prev = (pVolatileTransaction) p_txn;
489 pVolatileTransaction(p_txn)->p_prev = p_last;
490 p_last->p_next = (pVolatileTransaction) p_txn;
491
492 it_ent->second = (pVolatileTransaction) p_txn;
493
494 return SERVICE_NO_ERROR;
495 }
496 pVolatileTransaction p_last = it_ent->second->p_prev;
497
498 pVolatileTransaction(p_txn)->p_prev = p_last;
499 it_ent->second->p_prev = (pVolatileTransaction) p_txn;
500 pVolatileTransaction(p_txn)->p_next = it_ent->second;
501 p_last->p_next = (pVolatileTransaction) p_txn;
502
503 return SERVICE_NO_ERROR;
504 }

517 inline StatusCode put_in_tree(HashVolXctMap::iterator it_ent, EntityKeyHash &ek, Name &key, Name &parent, pBlock p_block) {
518
519 pVolatileTransaction p_root, p_parent = nullptr, p_next = nullptr;;
520
521 if ((p_root = it_ent->second) != nullptr) {
522 if (tree_key.find(ek) != tree_key.end())
523 return SERVICE_ERROR_WRITE_FORBIDDEN;
524
525 EntityKeyHash parent_ek = {ek.ent_hash, hash(parent)};
526
527 EntKeyVolXctMap::iterator it;
528
529 if ((it = tree_key.find(parent_ek)) == tree_key.end())
530 return SERVICE_ERROR_PARENT_NOT_FOUND;
531
532 p_parent = it->second;
533 p_next = p_parent->p_child;
534 }
535
536 pTransaction p_txn;
537 int ret;
538
539 if ((ret = new_transaction(p_txn)) != SERVICE_NO_ERROR)
540 return ret;
541
542 pBlock p_new = block_malloc(p_block->total_bytes);
543 if (p_new == nullptr) {
544 destroy_transaction(p_txn);
545
546 return SERVICE_ERROR_NO_MEM;
547 }
548 memcpy(p_new, p_block, p_block->total_bytes);
549 p_txn->p_block = p_new;
550 p_txn->p_block->hash64 = 0;
551 p_txn->status = BLOCK_STATUS_READY;
552
553 tree_key[ek] = (pVolatileTransaction) p_txn;
554
555 pVolatileTransaction(p_txn)->p_parent = p_parent;
556 pVolatileTransaction(p_txn)->p_child = nullptr;
557 pVolatileTransaction(p_txn)->p_next = p_next;
558 pVolatileTransaction(p_txn)->num_wins = 0;
559 pVolatileTransaction(p_txn)->num_visits = 0;
560 pVolatileTransaction(p_txn)->key_hash = add_name(ek.key_hash, key);
561
562 if (p_root == nullptr)
563 it_ent->second = (pVolatileTransaction) p_txn;
564 else
565 p_parent->p_child = (pVolatileTransaction) p_txn;
566
567 return SERVICE_NO_ERROR;
568 }

642 inline void destroy_item(int base, uint64_t ent_hash, pVolatileTransaction p_item) {
643
644 EntityKeyHash ek = {ent_hash, p_item->key_hash};
645 pTransaction p_txn = p_item;
646
647 switch (base) {
648 case BASE_DEQUE_10BIT: {
649 HashVolXctMap::iterator it_ent = deque_ent.find(ent_hash);
650 if (p_item == it_ent->second) {
651 if (p_item->p_next == p_item)
652 it_ent->second = nullptr;
653 else {
654 p_item->p_next->p_prev = p_item->p_prev;
655 p_item->p_prev->p_next = p_item->p_next;
656 it_ent->second = p_item->p_next;
657 }
658 } else {
659 p_item->p_next->p_prev = p_item->p_prev;
660 p_item->p_prev->p_next = p_item->p_next;
661 }
662 erase_name(ek.key_hash);
663 deque_key.erase(ek);
664 destroy_transaction(p_txn); }
665
666 return;
667
668 case BASE_QUEUE_10BIT: {
669 HashQueueEntMap::iterator it_ent = queue_ent.find(ent_hash);
670
671 it_ent->second.p_root = aat_remove(p_item, it_ent->second.p_root);
672 it_ent->second.queue_use--;
673
674 erase_name(ek.key_hash);
675 queue_key.erase(ek);
676 destroy_transaction(p_txn); }
677
678 return;
679
680 case BASE_TREE_10BIT:
681 erase_name(ek.key_hash);
682 destroy_transaction(p_txn);
683 tree_key.erase(ek);
684 }
685 }

695 inline uint64_t add_name(uint64_t hash, Name &key) {
696
697 HashNameUseMap::iterator it = name.find(hash);
698
699 if (it != name.end())
700 name[hash].use++;
701 else {
702 NameUse nu;
703
704 nu.use = 1;
705 memcpy(&nu.name, &key, sizeof(Name));
706
707 name[hash] = nu;
708 }
709 return hash;
710 }

718 inline void erase_name(uint64_t hash) {
719
720 HashNameUseMap::iterator it = name.find(hash);
721
722 if (it == name.end())
723 return;
724
725 if (--name[hash].use == 0)
726 name.erase(it);
727 }

742 inline StatusCode internal_get(pTransaction &p_txn, pString &p_str, uint64_t &pop_ent, Locator &what) {
743
744 int base;
745 pVolatileTransaction p_root;
746 EntityKeyHash ek;
747 ek.ent_hash = hash(what.entity);
748
749 pop_ent = 0;
750
751 switch (base = TenBitsAtAddress(what.base)) {
752 case BASE_DEQUE_10BIT: {
753 HashVolXctMap::iterator it_ent = deque_ent.find(ek.ent_hash);
754
755 if (it_ent == deque_ent.end())
756 return SERVICE_ERROR_ENTITY_NOT_FOUND;
757
758 p_root = it_ent->second; }
759
760 break;
761
762 case BASE_INDEX_10BIT: {
763 HashVolXctMap::iterator it_ent = index_ent.find(ek.ent_hash);
764
765 if (it_ent == index_ent.end())
766 return SERVICE_ERROR_ENTITY_NOT_FOUND;
767
768 p_root = it_ent->second; }
769
770 break;
771
772 case BASE_QUEUE_10BIT: {
773 HashQueueEntMap::iterator it_ent = queue_ent.find(ek.ent_hash);
774
775 if (it_ent == queue_ent.end())
776 return SERVICE_ERROR_ENTITY_NOT_FOUND;
777
778 p_root = it_ent->second.p_root; }
779
780 break;
781
782 case BASE_TREE_10BIT: {
783 HashVolXctMap::iterator it_ent = tree_ent.find(ek.ent_hash);
784
785 if (it_ent == tree_ent.end())
786 return SERVICE_ERROR_ENTITY_NOT_FOUND;
787
788 p_root = it_ent->second; }
789
790 break;
791
792 default:
793 return SERVICE_ERROR_WRONG_BASE;
794 }
795
796 if (p_root == nullptr)
797 return SERVICE_ERROR_EMPTY_ENTITY;
798
799 Name key, second;
800 int command;
801
802 if (!parse_command(key, command, second, what.key, false))
803 return SERVICE_ERROR_PARSING_COMMAND;
804
805 switch (command) {
806 case COMMAND_JUST_THE_KEY:
807 switch (base) {
808 case BASE_DEQUE_10BIT: {
809 ek.key_hash = hash(key);
810 EntKeyVolXctMap::iterator it;
811
812 if ((it = deque_key.find(ek)) == deque_key.end())
813 return SERVICE_ERROR_BLOCK_NOT_FOUND;
814
815 p_txn = it->second;
816 p_str = nullptr; }
817
818 return SERVICE_NO_ERROR;
819
820 case BASE_QUEUE_10BIT: {
821 ek.key_hash = hash(key);
822 EntKeyVolXctMap::iterator it;
823
824 if ((it = queue_key.find(ek)) == queue_key.end())
825 return SERVICE_ERROR_BLOCK_NOT_FOUND;
826
827 p_txn = it->second;
828 p_str = nullptr; }
829
830 return SERVICE_NO_ERROR;
831
832 case BASE_TREE_10BIT: {
833 ek.key_hash = hash(key);
834 EntKeyVolXctMap::iterator it;
835
836 if ((it = tree_key.find(ek)) == tree_key.end())
837 return SERVICE_ERROR_BLOCK_NOT_FOUND;
838
839 p_txn = it->second;
840 p_str = nullptr; }
841
842 return SERVICE_NO_ERROR;
843
844 default: {
845 Index::iterator it;
846
847 if ((it = p_root->p_hea->index.find(key)) == p_root->p_hea->index.end())
848 return SERVICE_ERROR_BLOCK_NOT_FOUND;
849
850 p_txn = nullptr;
851 p_str = &it->second; }
852
853 return SERVICE_NO_ERROR;
854 }
855
856 case COMMAND_CHILD_10BIT: {
857 if (base != BASE_TREE_10BIT)
858 return SERVICE_ERROR_PARSING_COMMAND;
859
860 ek.key_hash = hash(key);
861 EntKeyVolXctMap::iterator it;
862
863 if ((it = tree_key.find(ek)) == tree_key.end())
864 return SERVICE_ERROR_BLOCK_NOT_FOUND;
865
866 p_txn = it->second->p_child;
867
868 if (p_txn == nullptr)
869 return SERVICE_ERROR_BLOCK_NOT_FOUND;
870
871 p_str = nullptr; }
872
873 return SERVICE_NO_ERROR;
874
875 case COMMAND_PARENT_10BIT: {
876 if (base != BASE_TREE_10BIT)
877 return SERVICE_ERROR_PARSING_COMMAND;
878
879 ek.key_hash = hash(key);
880 EntKeyVolXctMap::iterator it;
881
882 if ((it = tree_key.find(ek)) == tree_key.end())
883 return SERVICE_ERROR_BLOCK_NOT_FOUND;
884
885 p_txn = it->second->p_parent;
886
887 if (p_txn == nullptr)
888 return SERVICE_ERROR_BLOCK_NOT_FOUND;
889
890 p_str = nullptr; }
891
892 return SERVICE_NO_ERROR;
893
894 case COMMAND_NEXT_10BIT: {
895 ek.key_hash = hash(key);
896 EntKeyVolXctMap::iterator it;
897
898 switch (base) {
899 case BASE_TREE_10BIT: {
900 if ((it = tree_key.find(ek)) == tree_key.end())
901 return SERVICE_ERROR_BLOCK_NOT_FOUND;
902
903 p_txn = it->second->p_next;
904
905 if (p_txn == nullptr)
906 return SERVICE_ERROR_BLOCK_NOT_FOUND;
907
908 p_str = nullptr; }
909
910 return SERVICE_NO_ERROR;
911
912 case BASE_DEQUE_10BIT: {
913 if ((it = deque_key.find(ek)) == deque_key.end())
914 return SERVICE_ERROR_BLOCK_NOT_FOUND;
915
916 p_txn = it->second->p_next;
917 p_str = nullptr; }
918
919 return SERVICE_NO_ERROR;
920 }
921 return SERVICE_ERROR_PARSING_COMMAND; }
922
923 case COMMAND_PREV_10BIT: {
924 if (base != BASE_DEQUE_10BIT)
925 return SERVICE_ERROR_PARSING_COMMAND;
926
927 ek.key_hash = hash(key);

1030 inline bool parse_command(Name &key_out, int &command, Name &second, Name key_in, bool is_put) {
1031 if (key_in[0] == '~') {
1032 key_out[0] = 0;
1033 second [0] = 0;
1034
1035 switch (command = TenBitsAtAddress(&key_in[1])) {
1036 case COMMAND_FIRST_10BIT:
1037 case COMMAND_LAST_10BIT:
1038 return true;
1039
1040 case COMMAND_PUT_10BIT:
1041 return is_put;
1042
1043 case COMMAND_PFIRST_10BIT:
1044 case COMMAND_PLAST_10BIT:
1045 case COMMAND_HIGH_10BIT:
1046 case COMMAND_LOW_10BIT:
1047 case COMMAND_GET_10BIT:
1048 case COMMAND_XHIGH_10BIT:
1049 case COMMAND_XLOW_10BIT:
1050 return !is_put;
1051
1052 default:
1053 if (!is_put || key_in[1] < '0' || key_in[1] > '9')
1054 return false;
1055
1056 int size, r_len;
1057
1058 if (sscanf(&key_in[1], "%d%n", &size, &r_len) != 1 || size <= 0 || key_in[r_len + 1] != 0)
1059 return false;
1060
1061 command = COMMAND_SIZE + size;
1062
1063 return true;
1064 }
1065 }
1066 strcpy(key_out, key_in);
1067 pChar pc = strchr(key_out, '~');
1068
1069 if (pc == nullptr) {
1070 command = COMMAND_JUST_THE_KEY;
1071 second[0] = 0;
1072
1073 return true;
1074 }
1075 (*pc++) = 0;
1076
1077 if (*pc == 0)
1078 return false;
1079
1080 if (is_put) {
1081 strcpy(second, pc);
1082 command = COMMAND_SECOND_ARG;
1083
1084 return true;
1085 }
1086 second[0] = 0;
1087
1088 switch (command = TenBitsAtAddress(pc)) {
1089 case COMMAND_CHILD_10BIT:
1090 case COMMAND_NEXT_10BIT:
1091 case COMMAND_PARENT_10BIT:
1092 case COMMAND_PREV_10BIT:
1093 return true;
1094 }
1095
1096 return false;
1097 }
