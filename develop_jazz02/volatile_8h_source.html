265  inline bool parse_command(Name &key_out, int &command, Name &parent, Name key_in, bool is_put) {
266  if (key_in[0] == '~') {
267  key_out[0] = 0;
268  parent [0] = 0;
269 
270  switch (command = TenBitsAtAddress(&key_in[1])) {
271  case COMMAND_CHILD_10BIT:
272  case COMMAND_FIRST_10BIT:
273  case COMMAND_GET_10BIT:
274  case COMMAND_HIGH_10BIT:
275  case COMMAND_II_10BIT:
276  case COMMAND_INSERT_10BIT:
277  case COMMAND_IS_10BIT:
278  case COMMAND_LAST_10BIT:
279  case COMMAND_LOW_10BIT:
280  case COMMAND_NEXT_10BIT:
281  case COMMAND_PARENT_10BIT:
282  case COMMAND_PFIRST_10BIT:
283  case COMMAND_PLAST_10BIT:
284  case COMMAND_PREV_10BIT:
285  case COMMAND_PUT_10BIT:
286  case COMMAND_SI_10BIT:
287  case COMMAND_SS_10BIT:
288  case COMMAND_XHIGH_10BIT:
289  case COMMAND_XLOW_10BIT:
290  return true;
291 
292  default:
293  int size;
294 
295  if (sscanf(&key_in[1], "%i", &size) != 1)
296  return false;
297 
298  command = COMMAND_SIZE + size;
299 
300  return true;
301  }
302  }
303  strcpy(key_out, key_in);
304  pChar pc = strchr(key_out, '~');
305 
306  if (pc == nullptr) {
307  command = COMMAND_JUST_THE_KEY;
308  parent[0] = 0;
309 
310  return true;
311  }
312  (*pc++) = 0;
313 
314  if (is_put) {
315  strcpy(parent, pc);
316  command = COMMAND_PARENT_KEY;
317 
318  return true;
319  }
320 
321  switch (command = TenBitsAtAddress(pc)) {
322  case COMMAND_CHILD_10BIT:
323  case COMMAND_NEXT_10BIT:
324  case COMMAND_PARENT_10BIT:
325  case COMMAND_PREV_10BIT:
326  return true;
327  }
328 
329  return false;
330  }

489  inline pVolatileTransaction remove_go_deep(pVolatileTransaction p_kill,
490  pVolatileTransaction p_parent,
491  pVolatileTransaction p_tree,
492  pVolatileTransaction &p_deep) {
493  if (p_kill->p_next != nullptr)
494  p_kill->p_next = remove_go_deep(p_kill->p_next, p_kill, p_tree, p_deep);
495 
496  else {
497  if (p_parent == p_tree) {
498  p_tree->p_prev = p_kill->p_prev; // Disconnect p_kill
499  p_kill->level = p_tree->level;
500  p_kill->p_next = p_tree->p_next;
501  p_kill->p_prev = p_tree->p_prev; // p_kill is the new p_tree
502 
503  return rebalance(p_kill);
504 
505  } else {
506  p_deep = p_kill; // Save p_kill for the end
507 
508  return p_kill->p_prev; // Disconnect p_kill
509  }
510  }
511  if (p_parent != p_tree)
512  return rebalance(p_kill);
513 
514  else {
515  p_deep->level = p_tree->level;
516  p_deep->p_next = p_tree->p_next;
517  p_deep->p_prev = rebalance(p_kill);
518 
519  return rebalance(p_deep);

533  inline pVolatileTransaction remove(pVolatileTransaction p_item, pVolatileTransaction p_tree) {
534 
535  if (p_tree == nullptr || p_item == nullptr)
536  return p_tree;
537 
538  if (p_item == p_tree) {
539  if (p_tree->p_prev == nullptr)
540  return p_tree->p_next;
541 
542  else {
543  if (p_tree->p_next == nullptr)
544  return p_tree->p_prev;
545 
546  else {
547  pVolatileTransaction p_deep = nullptr;
548  return remove_go_deep(p_tree->p_prev, p_tree, p_tree, p_deep);
549  }
550  }
551  } else {
552  if (to_left(p_item, p_tree))
553  p_tree->p_prev = remove(p_item, p_tree->p_prev);
554 
555  else
556  p_tree->p_next = remove(p_item, p_tree->p_next);
557  }
558 
559  return rebalance(p_tree);
560  };

569  inline void decrease_level(pVolatileTransaction p_item) {
570 
571  if (p_item->p_prev == nullptr) {
572  if (p_item->p_next == nullptr)
573  p_item->level = 1;
574 
575  else {
576  int should_be = p_item->p_next->level + 1;
577 
578  if (should_be < p_item->level) {
579  p_item->level = should_be;
580 
581  if (should_be < p_item->p_next->level)
582  p_item->p_next->level = should_be;
583  }
584  }
585  } else {
586  if (p_item->p_next == nullptr) {
587  int should_be = p_item->p_prev->level + 1;
588 
589  if (should_be < p_item->level)
590  p_item->level = should_be;
591 
592  } else {
593  int should_be = std::min(p_item->p_prev->level, p_item->p_next->level) + 1;
594 
595  if (should_be < p_item->level) {
596  p_item->level = should_be;
597 
598  if (should_be < p_item->p_next->level)
599  p_item->p_next->level = should_be;
600  }
601  }
602  }

636  inline pVolatileTransaction split(pVolatileTransaction p_item) {
637  // rotate p_prev if there are two p_next children on same level
638 
639  if ( p_item->p_next != nullptr
640  && p_item->p_next->p_next != nullptr
641  && p_item->level == p_item->p_next->p_next->level) {
642 
643  pVolatileTransaction p_right = p_item->p_next;
644 
645  p_item->p_next = p_right->p_prev;
646  p_right->p_prev = p_item;

654  // && p_right->p_next->p_next->level >= 2))

675  inline pVolatileTransaction insert(pVolatileTransaction p_new, pVolatileTransaction p_tree) {
676  // Do the normal binary tree insertion procedure. Set the result of the
677  // recursive call to the correct child in case a new node was created or the
678  // root of the subtree changes.
679 
680  if (p_tree == nullptr) {
681  p_new->level = 1;
682  p_new->p_prev = nullptr;
683  p_new->p_next = nullptr;
684 
685  return p_new;
686 
687  } else {
688 
689  if (to_left(p_new, p_tree))
690  p_tree->p_prev = insert(p_new, p_tree->p_prev);
691 
692  else
693  p_tree->p_next = insert(p_new, p_tree->p_next);
694  }
695 
696  // Perform skew and then split. The conditionals that determine whether or
697  // not a rotation will occur or not are inside of the procedures, as given above.
698 
699  p_tree = skew(p_tree);
700  p_tree = split(p_tree);
701 
702  return p_tree;
703  };
