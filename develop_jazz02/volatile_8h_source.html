294  inline StatusCode internal_get(pTransaction &p_txn, pString &p_str, uint64_t &pop_ent, Locator &what) {
295 
296  HashVolXctMap *p_ent_map;
297  int base;
298 
299  pop_ent = 0;
300 
301  switch (base = TenBitsAtAddress(what.base)) {
302  case BASE_DEQUE_10BIT:
303  p_ent_map = &deque_ent;
304  break;
305 
306  case BASE_INDEX_10BIT:
307  p_ent_map = &index_ent;
308  break;
309 
310  case BASE_QUEUE_10BIT:
311  p_ent_map = &queue_ent;
312  break;
313 
314  case BASE_TREE_10BIT:
315  p_ent_map = &tree_ent;
316  break;
317 
318  default:
319  return SERVICE_ERROR_WRONG_BASE;
320  }
321 
322  EntityKeyHash ek;
323  ek.ent_hash = hash(what.entity);
324 
325  HashVolXctMap::iterator it_ent = p_ent_map->find(ek.ent_hash);
326 
327  if (it_ent == p_ent_map->end())
328  return SERVICE_ERROR_BLOCK_NOT_FOUND;
329 
330  pVolatileTransaction p_root = it_ent->second;
331 
332  Name key, parent;
333  int command;
334 
335  if (!parse_command(key, command, parent, what.key, false))
336  return SERVICE_ERROR_PARSING_COMMAND;
337 
338  switch (command) {
339  case COMMAND_JUST_THE_KEY:
340  switch (base) {
341  case BASE_DEQUE_10BIT: {
342  ek.key_hash = hash(key);
343  EntKeyVolXctMap::iterator it;
344 
345  if ((it = deque_key.find(ek)) == deque_key.end())
346  return SERVICE_ERROR_BLOCK_NOT_FOUND;
347 
348  p_txn = it->second;
349  p_str = nullptr; }
350 
351  return SERVICE_NO_ERROR;
352 
353  case BASE_QUEUE_10BIT: {
354  ek.key_hash = hash(key);
355  EntKeyVolXctMap::iterator it;
356 
357  if ((it = queue_key.find(ek)) == queue_key.end())
358  return SERVICE_ERROR_BLOCK_NOT_FOUND;
359 
360  p_txn = it->second;
361  p_str = nullptr; }
362 
363  return SERVICE_NO_ERROR;
364 
365  case BASE_TREE_10BIT: {
366  ek.key_hash = hash(key);
367  EntKeyVolXctMap::iterator it;
368 
369  if ((it = tree_key.find(ek)) == tree_key.end())
370  return SERVICE_ERROR_BLOCK_NOT_FOUND;
371 
372  p_txn = it->second;
373  p_str = nullptr; }
374 
375  return SERVICE_NO_ERROR;
376 
377  default: {
378  Index::iterator it;
379 
380  if ((it = p_root->p_hea->index.find(key)) == p_root->p_hea->index.end())
381  return SERVICE_ERROR_BLOCK_NOT_FOUND;
382 
383  p_txn = nullptr;
384  p_str = &it->second; }
385 
386  return SERVICE_NO_ERROR;
387  }
388 
389  case COMMAND_CHILD_10BIT: {
390  if (base != BASE_TREE_10BIT)
391  return SERVICE_ERROR_BLOCK_NOT_FOUND;
392 
393  ek.key_hash = hash(key);
394  EntKeyVolXctMap::iterator it;
395 
396  if ((it = tree_key.find(ek)) == tree_key.end())
397  return SERVICE_ERROR_BLOCK_NOT_FOUND;
398 
399  p_txn = it->second->p_child;
400  p_str = nullptr; }
401 
402  return SERVICE_NO_ERROR;
403 
404  case COMMAND_PARENT_10BIT: {
405  if (base != BASE_TREE_10BIT)
406  return SERVICE_ERROR_BLOCK_NOT_FOUND;
407 
408  ek.key_hash = hash(key);
409  EntKeyVolXctMap::iterator it;
410 
411  if ((it = tree_key.find(ek)) == tree_key.end())
412  return SERVICE_ERROR_BLOCK_NOT_FOUND;
413 
414  p_txn = it->second->p_parent;
415  p_str = nullptr; }
416 
417  return SERVICE_NO_ERROR;
418 
419  case COMMAND_NEXT_10BIT: {
420  if (base != BASE_TREE_10BIT && base != BASE_DEQUE_10BIT)
421  return SERVICE_ERROR_BLOCK_NOT_FOUND;
422 
423  ek.key_hash = hash(key);
424  EntKeyVolXctMap::iterator it;
425 
426  if ((it = tree_key.find(ek)) == tree_key.end())
427  return SERVICE_ERROR_BLOCK_NOT_FOUND;
428 
429  p_txn = it->second->p_next;
430  p_str = nullptr; }
431 
432  return SERVICE_NO_ERROR;
433 
434  case COMMAND_PREV_10BIT: {
435  if (base != BASE_DEQUE_10BIT)
436  return SERVICE_ERROR_BLOCK_NOT_FOUND;
437 
438  ek.key_hash = hash(key);
439  EntKeyVolXctMap::iterator it;
440 
441  if ((it = tree_key.find(ek)) == tree_key.end())
442  return SERVICE_ERROR_BLOCK_NOT_FOUND;
443 
444  p_txn = it->second->p_prev;
445  p_str = nullptr; }
446 
447  return SERVICE_NO_ERROR;
448 
449  case COMMAND_HIGH_10BIT:
450  case COMMAND_XHIGH_10BIT: {
451  if (base != BASE_QUEUE_10BIT)
452  return SERVICE_ERROR_BLOCK_NOT_FOUND;
453 
454  if (p_root == nullptr)
455  return SERVICE_ERROR_EMPTY_ENTITY;
456 
457  p_txn = highest_priority(p_root);
458  p_str = nullptr;
459 
460  if (command == COMMAND_XHIGH_10BIT)
461  pop_ent = ek.ent_hash; }
462 
463  return SERVICE_NO_ERROR;
464 
465  case COMMAND_LOW_10BIT:
466  case COMMAND_XLOW_10BIT: {
467  if (base != BASE_QUEUE_10BIT)
468  return SERVICE_ERROR_BLOCK_NOT_FOUND;
469 
470  if (p_root == nullptr)
471  return SERVICE_ERROR_EMPTY_ENTITY;
472 
473  p_txn = lowest_priority(p_root);
474  p_str = nullptr;
475 
476  if (command == COMMAND_XLOW_10BIT)
477  pop_ent = ek.ent_hash; }
478 
479  return SERVICE_NO_ERROR;
480 
481  case COMMAND_FIRST_10BIT:
482  case COMMAND_PFIRST_10BIT: {
483  if (base != BASE_DEQUE_10BIT && (base != BASE_TREE_10BIT || command == COMMAND_PFIRST_10BIT))
484  return SERVICE_ERROR_BLOCK_NOT_FOUND;
485 
486  if (p_root == nullptr)
487  return SERVICE_ERROR_EMPTY_ENTITY;
488 
489  p_txn = p_root;
490  p_str = nullptr;
491 
492  if (command == COMMAND_PFIRST_10BIT)
493  pop_ent = ek.ent_hash; }
494 
495  return SERVICE_NO_ERROR;
496 
497  case COMMAND_LAST_10BIT:
498  case COMMAND_PLAST_10BIT: {
499  if (base != BASE_DEQUE_10BIT)
500  return SERVICE_ERROR_BLOCK_NOT_FOUND;
501 
502  if (p_root == nullptr)
503  return SERVICE_ERROR_EMPTY_ENTITY;
504 
505  p_txn = p_root->p_prev;
506  p_str = nullptr;
507 
508  if (command == COMMAND_PLAST_10BIT)
509  pop_ent = ek.ent_hash; }
510 
511  return SERVICE_NO_ERROR;
512 
513  case COMMAND_GET_10BIT:
514  if (base != BASE_INDEX_10BIT)
515  return SERVICE_ERROR_BLOCK_NOT_FOUND;
516 
517  p_txn = nullptr;
518  p_str = nullptr;
519  pop_ent = ek.ent_hash;
520 
521  return SERVICE_NO_ERROR;
522  }
523 
524  return SERVICE_ERROR_PARSING_COMMAND;
525  }

540  inline bool parse_command(Name &key_out, int &command, Name &parent, Name key_in, bool is_put) {
541  if (key_in[0] == '~') {
542  key_out[0] = 0;
543  parent [0] = 0;
544 
545  switch (command = TenBitsAtAddress(&key_in[1])) {
546  case COMMAND_FIRST_10BIT:
547  case COMMAND_LAST_10BIT:
548  return true;
549 
550  case COMMAND_PUT_10BIT:
551  case COMMAND_INSERT_10BIT:
552  return is_put;
553 
554  case COMMAND_II_10BIT:
555  case COMMAND_IS_10BIT:
556  case COMMAND_SI_10BIT:
557  case COMMAND_SS_10BIT:
558  case COMMAND_PFIRST_10BIT:
559  case COMMAND_PLAST_10BIT:
560  case COMMAND_HIGH_10BIT:
561  case COMMAND_LOW_10BIT:
562  case COMMAND_GET_10BIT:
563  case COMMAND_XHIGH_10BIT:
564  case COMMAND_XLOW_10BIT:
565  return !is_put;
566 
567  default:
568  if (!is_put || key_in[1] < '0' || key_in[1] > '9')
569  return false;
570 
571  int size, r_len;
572 
573  if (sscanf(&key_in[1], "%d%n", &size, &r_len) != 1 || size <= 0 || key_in[r_len + 1] != 0)
574  return false;
575 
576  command = COMMAND_SIZE + size;
577 
578  return true;
579  }
580  }
581  strcpy(key_out, key_in);
582  pChar pc = strchr(key_out, '~');
583 
584  if (pc == nullptr) {
585  command = COMMAND_JUST_THE_KEY;
586  parent[0] = 0;
587 
588  return true;
589  }
590  (*pc++) = 0;
591 
592  if (*pc == 0)
593  return false;
594 
595  if (is_put) {
596  if (!valid_name(pc))
597  return false;
598 
599  strcpy(parent, pc);
600  command = COMMAND_PARENT_KEY;
601 
602  return true;
603  }
604  parent[0] = 0;
605 
606  switch (command = TenBitsAtAddress(pc)) {
607  case COMMAND_CHILD_10BIT:
608  case COMMAND_NEXT_10BIT:
609  case COMMAND_PARENT_10BIT:
610  case COMMAND_PREV_10BIT:
611  return true;
612  }
613 
614  return false;
615  }

774  inline pVolatileTransaction remove_go_deep(pVolatileTransaction p_kill,
775  pVolatileTransaction p_parent,
776  pVolatileTransaction p_tree,
777  pVolatileTransaction &p_deep) {
778  if (p_kill->p_next != nullptr)
779  p_kill->p_next = remove_go_deep(p_kill->p_next, p_kill, p_tree, p_deep);
780 
781  else {
782  if (p_parent == p_tree) {
783  p_tree->p_prev = p_kill->p_prev; // Disconnect p_kill
784  p_kill->level = p_tree->level;
785  p_kill->p_next = p_tree->p_next;
786  p_kill->p_prev = p_tree->p_prev; // p_kill is the new p_tree
787 
788  return rebalance(p_kill);
789 
790  } else {
791  p_deep = p_kill; // Save p_kill for the end
792 
793  return p_kill->p_prev; // Disconnect p_kill
794  }
795  }
796  if (p_parent != p_tree)
797  return rebalance(p_kill);
798 
799  else {
800  p_deep->level = p_tree->level;
801  p_deep->p_next = p_tree->p_next;
802  p_deep->p_prev = rebalance(p_kill);
803 
804  return rebalance(p_deep);
805  }

818  inline pVolatileTransaction remove(pVolatileTransaction p_item, pVolatileTransaction p_tree) {
819 
820  if (p_tree == nullptr || p_item == nullptr)
821  return p_tree;
822 
823  if (p_item == p_tree) {
824  if (p_tree->p_prev == nullptr)
825  return p_tree->p_next;
826 
827  else {
828  if (p_tree->p_next == nullptr)
829  return p_tree->p_prev;
830 
831  else {
832  pVolatileTransaction p_deep = nullptr;
833  return remove_go_deep(p_tree->p_prev, p_tree, p_tree, p_deep);

836  } else {

838  p_tree->p_prev = remove(p_item, p_tree->p_prev);
839 
840  else
841  p_tree->p_next = remove(p_item, p_tree->p_next);
842  }
843 
844  return rebalance(p_tree);
845  };

854  inline void decrease_level(pVolatileTransaction p_item) {
855 
856  if (p_item->p_prev == nullptr) {
857  if (p_item->p_next == nullptr)
858  p_item->level = 1;
859 
860  else {

862 
863  if (should_be < p_item->level) {
864  p_item->level = should_be;
865 
866  if (should_be < p_item->p_next->level)
867  p_item->p_next->level = should_be;
868  }
869  }
870  } else {
871  if (p_item->p_next == nullptr) {
872  int should_be = p_item->p_prev->level + 1;
873 
874  if (should_be < p_item->level)
875  p_item->level = should_be;
876 
877  } else {
878  int should_be = std::min(p_item->p_prev->level, p_item->p_next->level) + 1;
879 
880  if (should_be < p_item->level) {
881  p_item->level = should_be;
882 
883  if (should_be < p_item->p_next->level)
884  p_item->p_next->level = should_be;
885  }
886  }
