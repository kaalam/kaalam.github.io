198  switch (set_has_NA) {
199  case JAZZ_SET_HAS_NA_FALSE:
200  p_block->has_NA = false;

202  case JAZZ_SET_HAS_NA_TRUE:
203  p_block->has_NA = p_block->cell_type != CELL_TYPE_BYTE; // CELL_TYPE_BYTE must always be has_NA == FALSE

206  p_block->has_NA = p_block->find_NAs_in_tensor();

307  inline void enter_reading(JazzLock &_lock_) {
308  int retry = 0;
309  while (true) {
310  int32_t lock = _lock_.fetch_add(1, std::memory_order_relaxed);
311  if (lock >= 0)
312  return;
313  _lock_.fetch_sub(1, std::memory_order_relaxed);
314

316  retry = 0;

511  inline void decrease_level(pJazzQueueItem p_item)
512  {
513  if ((p_item->p_alloc_prev != nullptr) & (p_item->p_alloc_next != nullptr)) {
514  int should_be = std::min(reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level,
515  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level + 1);
516 
517  if (should_be < p_item->level) {
518  p_item->level = should_be;
519  if (should_be < reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level) {
520  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level = should_be;
521  }
522  }
523  }
524  };

533  inline pJazzQueueItem skew(pJazzQueueItem p_item)
534  {
535  // rotate p_alloc_next if p_alloc_prev child has same level
536  if ((p_item->p_alloc_prev != nullptr) & (p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level)) {
537  pJazzQueueItem p_left = (pJazzQueueItem) p_item->p_alloc_prev;
538 
539  p_item->p_alloc_prev = p_left->p_alloc_next;
540  p_left->p_alloc_next = p_item;
541 
542  return p_left;
543  }
544 
545  return p_item;
546  };
547 
555  inline pJazzQueueItem split(pJazzQueueItem p_item)
556  {
557  // rotate p_alloc_prev if there are two p_alloc_next children on same level
558  if ( (p_item->p_alloc_next != nullptr)
559  & (p_item->p_alloc_next->p_alloc_next != nullptr)
560  & (p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next->p_alloc_next)->level)) {
561 
562  pJazzQueueItem p_right = (pJazzQueueItem) p_item->p_alloc_next;
563 
564  p_item->p_alloc_next = p_right->p_alloc_prev;
565  p_right->p_alloc_prev = p_item;
566  p_right->level++;
567 
568  return p_right;
569  }
570 
571  return p_item;
572  };
