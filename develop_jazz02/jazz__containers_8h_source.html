420  inline void enter_reading() {
421  int retry = 0;
422  while (true) {
423  int32_t lock = _keepr_lock_;
424  if (lock >= 0) {
425  int32_t next = lock + 1;
426  if (_keepr_lock_.compare_exchange_weak(lock, next))
427  return;
428  }
429  if (++retry > JAZZ_LOCK_READING_RETRY_NUMTIMES) {
430  std::this_thread::yield();
431  retry = 0;
432  }
433  }
434  }
435 
441  inline void leave_reading() {
442  while (true) {
443  int32_t lock = _keepr_lock_;
444  int32_t next = lock - 1;
445  if (_keepr_lock_.compare_exchange_weak(lock, next))
446  return;
447  }
448  }
449 
458  inline void enter_writing() {
459  int retry = 0;
460  while (true) {
461  int32_t lock = _keepr_lock_;
462  if (lock >= 0) {
463  int32_t next = lock - JAZZ_LOCK_WEIGHT_OF_WRITE;
464  if (_keepr_lock_.compare_exchange_weak(lock, next)) {
465  while (true) {
466  if (_keepr_lock_ == -JAZZ_LOCK_WEIGHT_OF_WRITE)
467  return;
468  if (++retry > JAZZ_LOCK_KICKING_RETRY_NUMTIMES) {
469  std::this_thread::yield();
470  retry = 0;
471  }
472  }
473  }
474  }
475  if (++retry > JAZZ_LOCK_WRITING_RETRY_NUMTIMES) {
476  std::this_thread::yield();
477  retry = 0;
478  }
479  }
480  }
481 
486  inline void leave_writing() {
487  while (true) {
488  int32_t lock = _keepr_lock_;
489  int32_t next = lock + JAZZ_LOCK_WEIGHT_OF_WRITE;
490  if (_keepr_lock_.compare_exchange_weak(lock, next))
491  return;
492  }
493  }

774  inline pJazzQueueItem remove_go_deep(pJazzQueueItem p_kill, pJazzQueueItem p_parent, pJazzQueueItem p_tree, pJazzQueueItem &p_deep) {
775  if (p_kill->p_alloc_next != nullptr)
776  p_kill->p_alloc_next = remove_go_deep((pJazzQueueItem) p_kill->p_alloc_next, p_kill, p_tree, p_deep);
777  else {
778  if (p_parent == p_tree) {
779  p_tree->p_alloc_prev = p_kill->p_alloc_prev; // Disconnect p_kill
780  p_kill->level = p_tree->level;
781  p_kill->p_alloc_next = p_tree->p_alloc_next;
782  p_kill->p_alloc_prev = p_tree->p_alloc_prev; // p_kill is the new p_tree
783  return rebalance(p_kill);
784  } else {
785  p_deep = p_kill; // Save p_kill for the end
786  return (pJazzQueueItem) p_kill->p_alloc_prev; // Disconnect p_kill
787  }
788  }
789  if (p_parent != p_tree)
790  return rebalance(p_kill);
791  else {
792  p_deep->level = p_tree->level;
793  p_deep->p_alloc_next = p_tree->p_alloc_next;
794  p_deep->p_alloc_prev = rebalance(p_kill);
795 
796  return rebalance(p_deep);
797  }

808  inline pJazzQueueItem remove(pJazzQueueItem p_item, pJazzQueueItem p_tree) {
809  if (p_tree == nullptr || p_item == nullptr)
810  return p_tree;
811 
812  if (p_item == p_tree) {
813  if (p_tree->p_alloc_prev == nullptr) {
814  return (pJazzQueueItem) p_tree->p_alloc_next;
815  } else {
816  if (p_tree->p_alloc_next == nullptr)
817  return (pJazzQueueItem) p_tree->p_alloc_prev;
818  else {
819  pJazzQueueItem p_deep = nullptr;
820  return remove_go_deep((pJazzQueueItem) p_tree->p_alloc_prev, p_tree, p_tree, p_deep);
821  }
822  }
823  } else {
824  if (to_left(p_item, p_tree))
825  p_tree->p_alloc_prev = remove(p_item, (pJazzQueueItem) p_tree->p_alloc_prev);
826  else
827  p_tree->p_alloc_next = remove(p_item, (pJazzQueueItem) p_tree->p_alloc_next);
828  }
829  return rebalance(p_tree);
830  };

838  inline void decrease_level(pJazzQueueItem p_item) {
839  if (p_item->p_alloc_prev == nullptr) {
840  if (p_item->p_alloc_next == nullptr) {
841  p_item->level = 1;
842  } else {
843  int should_be = reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level + 1;
844  if (should_be < p_item->level) {
845  p_item->level = should_be;
846  if (should_be < reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level)
847  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level = should_be;
848  }
849  }
850  } else {
851  if (p_item->p_alloc_next == nullptr) {
852  int should_be = reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level + 1;
853 
854  if (should_be < p_item->level)
855  p_item->level = should_be;
856  } else {
857  int should_be = std::min(reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level,
858  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level) + 1;
859 
860  if (should_be < p_item->level) {
861  p_item->level = should_be;
862  if (should_be < reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level)
863  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level = should_be;
864  }
865  }
866  }
867  };

876  inline pJazzQueueItem skew(pJazzQueueItem p_item) {
877  // rotate p_alloc_next if p_alloc_prev child has same level
878  if (p_item->p_alloc_prev != nullptr && p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level) {
879  pJazzQueueItem p_left = (pJazzQueueItem) p_item->p_alloc_prev;
880 
881  p_item->p_alloc_prev = p_left->p_alloc_next;
882  p_left->p_alloc_next = p_item;
883 
884  return p_left;
885  }
886 
887  return p_item;
888  };

897  inline pJazzQueueItem split(pJazzQueueItem p_item) {
898  // rotate p_alloc_prev if there are two p_alloc_next children on same level
899  if ( p_item->p_alloc_next != nullptr
900  && p_item->p_alloc_next->p_alloc_next != nullptr
901  && p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next->p_alloc_next)->level) {
902 
903  pJazzQueueItem p_right = (pJazzQueueItem) p_item->p_alloc_next;
904 
905  p_item->p_alloc_next = p_right->p_alloc_prev;
906  p_right->p_alloc_prev = p_item;
907  if (p_item->p_alloc_prev == nullptr && p_item->p_alloc_next == nullptr && p_item->level > 1) {
908  p_item->level = 1;
909 /* The following condition looks like it could be problematic (since the previous one is already an undocumented deviation from the
910 canonical method description). It has never been observed (in intense unit testing) and should only be considered if problems happen.
911 It may very well be impossible, who knows. Just keep it as a remark, unless someone smarter proves it unnecessary.
912  if ( p_right->level != 2
913  || ( p_right->p_alloc_next->p_alloc_next != nullptr
914  && reinterpret_cast<pJazzQueueItem>(p_right->p_alloc_next->p_alloc_next)->level >= 2)) {
915  p_right->level++; // This is actually a breakpoint possibility, not a solution!!
916  } */
917  } else {
918  p_right->level++;
919  }
920 
921  return p_right;
922  }
923 
924  return p_item;
925  };

935  inline pJazzQueueItem insert(pJazzQueueItem p_new, pJazzQueueItem p_tree) {
936  // Do the normal binary tree insertion procedure. Set the result of the
937  // recursive call to the correct child in case a new node was created or the
938  // root of the subtree changes.
939 
940  if (p_tree == nullptr) {
941  p_new->level = 1;
942  p_new->p_alloc_prev = nullptr;
943  p_new->p_alloc_next = nullptr;
944 
945  return p_new;
946  } else {
947 
948  if (to_left(p_new, p_tree))
949  p_tree->p_alloc_prev = insert(p_new, (pJazzQueueItem) p_tree->p_alloc_prev);
950  else
951  p_tree->p_alloc_next = insert(p_new, (pJazzQueueItem) p_tree->p_alloc_next);
952  }
953 
954  // Perform skew and then split. The conditionals that determine whether or
955  // not a rotation will occur or not are inside of the procedures, as given above.
956 
957  p_tree = skew(p_tree);
958  p_tree = split(p_tree);
959 
960  return p_tree;
961  };
