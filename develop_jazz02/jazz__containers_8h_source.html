291  inline void enter_reading(JazzLock &_lock_) {
292  int retry = 0;
293  while (true) {
294  int32_t lock = _lock_.fetch_add(1, std::memory_order_relaxed);
295  if (lock >= 0)

495  inline void decrease_level(pJazzQueueItem p_item)
496  {
497  if ((p_item->p_alloc_prev != nullptr) & (p_item->p_alloc_next != nullptr)) {
498  int should_be = std::min(reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level,
499  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level + 1);
500 
501  if (should_be < p_item->level) {
502  p_item->level = should_be;
503  if (should_be < reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level) {
504  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level = should_be;
505  }
506  }
507  }

517  inline pJazzQueueItem skew(pJazzQueueItem p_item)
518  {
519  // rotate p_alloc_next if p_alloc_prev child has same level
520  if ((p_item->p_alloc_prev != nullptr) & (p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level)) {
521  pJazzQueueItem p_left = (pJazzQueueItem) p_item->p_alloc_prev;
522 
523  p_item->p_alloc_prev = p_left->p_alloc_next;
524  p_left->p_alloc_next = p_item;
525 
526  return p_left;
527  }
528 
529  return p_item;
530  };
531 
539  inline pJazzQueueItem split(pJazzQueueItem p_item)
540  {
541  // rotate p_alloc_prev if there are two p_alloc_next children on same level
542  if ( (p_item->p_alloc_next != nullptr)
543  & (p_item->p_alloc_next->p_alloc_next != nullptr)
544  & (p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next->p_alloc_next)->level)) {
545 
546  pJazzQueueItem p_right = (pJazzQueueItem) p_item->p_alloc_next;
547 
548  p_item->p_alloc_next = p_right->p_alloc_prev;
549  p_right->p_alloc_prev = p_item;
550  p_right->level++;
551 
552  return p_right;
553  }
554 
555  return p_item;
556  };
