387  inline void decrease_level(pJazzQueueItem p_item)
388  {
389  if ((p_item->p_alloc_prev != nullptr) & (p_item->p_alloc_next != nullptr)) {
390  int should_be = std::min(reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level,
391  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level + 1);
392 
393  if (should_be < p_item->level) {
394  p_item->level = should_be;
395  if (should_be < reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level) {
396  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level = should_be;
397  }
398  }
399  }

409  inline pJazzQueueItem skew(pJazzQueueItem p_item)
410  {
411  // rotate p_alloc_next if p_alloc_prev child has same level
412  if ((p_item->p_alloc_prev != nullptr) & (p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level)) {
413  pJazzQueueItem p_left = (pJazzQueueItem) p_item->p_alloc_prev;
414 
415  p_item->p_alloc_prev = p_left->p_alloc_next;
416  p_left->p_alloc_next = p_item;
417 
418  return p_left;
419  }
420 
421  return p_item;
422  };

431  inline pJazzQueueItem split(pJazzQueueItem p_item)
432  {
433  // rotate p_alloc_prev if there are two p_alloc_next children on same level
434  if ( (p_item->p_alloc_next != nullptr)
435  & (p_item->p_alloc_next->p_alloc_next != nullptr)
436  & (p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next->p_alloc_next)->level)) {
437 
438  pJazzQueueItem p_right = (pJazzQueueItem) p_item->p_alloc_next;
439 
440  p_item->p_alloc_next = p_right->p_alloc_prev;
441  p_right->p_alloc_prev = p_item;
442  p_right->level++;
443 
444  return p_right;
445  }
446 
447  return p_item;
448  };
