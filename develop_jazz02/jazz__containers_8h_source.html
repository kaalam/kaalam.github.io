323  int retry = 0;

526  inline void decrease_level(pJazzQueueItem p_item)
527  {
528  if ((p_item->p_alloc_prev != nullptr) & (p_item->p_alloc_next != nullptr)) {
529  int should_be = std::min(reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level,
530  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level + 1);
531 
532  if (should_be < p_item->level) {
533  p_item->level = should_be;
534  if (should_be < reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level) {
535  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level = should_be;
536  }
537  }
538  }
539  };

548  inline pJazzQueueItem skew(pJazzQueueItem p_item)
549  {
550  // rotate p_alloc_next if p_alloc_prev child has same level
551  if ((p_item->p_alloc_prev != nullptr) & (p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level)) {
552  pJazzQueueItem p_left = (pJazzQueueItem) p_item->p_alloc_prev;
553 
554  p_item->p_alloc_prev = p_left->p_alloc_next;
555  p_left->p_alloc_next = p_item;
556 
557  return p_left;
558  }
559 
560  return p_item;
561  };
562 
570  inline pJazzQueueItem split(pJazzQueueItem p_item)
571  {
572  // rotate p_alloc_prev if there are two p_alloc_next children on same level
573  if ( (p_item->p_alloc_next != nullptr)
574  & (p_item->p_alloc_next->p_alloc_next != nullptr)
575  & (p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next->p_alloc_next)->level)) {
576 
577  pJazzQueueItem p_right = (pJazzQueueItem) p_item->p_alloc_next;
578 
579  p_item->p_alloc_next = p_right->p_alloc_prev;
580  p_right->p_alloc_prev = p_item;
581  p_right->level++;
582 
583  return p_right;
584  }
585 
586  return p_item;
587  };
