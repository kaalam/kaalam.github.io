343  inline void enter_reading(JazzLock &_lock_) {
344  int retry = 0;
345  while (true) {
346  int lock = _lock_.fetch_add(1, std::memory_order_relaxed);
347  if (lock >= 0)
348  return;
349  _lock_.fetch_sub(1, std::memory_order_relaxed);
350 
351  if (++retry > JAZZ_LOCK_READING_RETRY_NUMTIMES) {
352  std::this_thread::yield();
353  retry = 0;
354  }
355  }
356  }
357 
365  inline void leave_reading(JazzLock &_lock_) { _lock_.fetch_sub(1, std::memory_order_relaxed); }
366 
377  inline void enter_writing(JazzLock &_lock_) {
378  int retry = 0;
379  while (true) {
380  int lock = _lock_.fetch_sub(JAZZ_LOCK_WEIGHT_OF_WRITE, std::memory_order_relaxed);
381  if (lock == 0)
382  return;
383  if (lock > 0) {
384  retry = 0;
385  while (true) {
386  if (_lock_ == 0)
387  return;
388  if (++retry > JAZZ_LOCK_KICKING_RETRY_NUMTIMES) {
389  std::this_thread::yield();
390  retry = 0;
391  }
392  }
393  }
394  _lock_.fetch_add(JAZZ_LOCK_WEIGHT_OF_WRITE, std::memory_order_relaxed);
395 
396  if (++retry > JAZZ_LOCK_WRITING_RETRY_NUMTIMES) {
397  std::this_thread::yield();
398  retry = 0;
399  }
400  }
401  }

580  inline pJazzQueueItem skew(pJazzQueueItem p_item)
581  {

602  inline pJazzQueueItem split(pJazzQueueItem p_item)
603  {
604  // rotate p_alloc_prev if there are two p_alloc_next children on same level
605  if ( (p_item->p_alloc_next != nullptr)

612  p_right->p_alloc_prev = p_item;
613  p_right->level++;
614
