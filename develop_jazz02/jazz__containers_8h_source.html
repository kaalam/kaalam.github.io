484  inline void decrease_level(pJazzQueueItem p_item)
485  {
486  if ((p_item->p_alloc_prev != nullptr) & (p_item->p_alloc_next != nullptr)) {
487  int should_be = std::min(reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level,
488  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level + 1);

492  if (should_be < reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level) {
493  reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next)->level = should_be;

506  inline pJazzQueueItem skew(pJazzQueueItem p_item)
507  {
508  // rotate p_alloc_next if p_alloc_prev child has same level
509  if ((p_item->p_alloc_prev != nullptr) & (p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_prev)->level)) {
510  pJazzQueueItem p_left = (pJazzQueueItem) p_item->p_alloc_prev;
511 
512  p_item->p_alloc_prev = p_left->p_alloc_next;
513  p_left->p_alloc_next = p_item;
514 
515  return p_left;
516  }
517 
518  return p_item;
519  };

528  inline pJazzQueueItem split(pJazzQueueItem p_item)
529  {
530  // rotate p_alloc_prev if there are two p_alloc_next children on same level
531  if ( (p_item->p_alloc_next != nullptr)
532  & (p_item->p_alloc_next->p_alloc_next != nullptr)
533  & (p_item->level == reinterpret_cast<pJazzQueueItem>(p_item->p_alloc_next->p_alloc_next)->level)) {
534 
535  pJazzQueueItem p_right = (pJazzQueueItem) p_item->p_alloc_next;
536 
537  p_item->p_alloc_next = p_right->p_alloc_prev;
538  p_right->p_alloc_prev = p_item;
539  p_right->level++;
540 
541  return p_right;
542  }
