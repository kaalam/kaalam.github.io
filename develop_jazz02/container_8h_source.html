527  inline bool get_item_name(pChar &p_in, int &num_bytes, pChar p_out, bool check_quotes = true, bool check_colon = true) {
528  if (skip_space(p_in, num_bytes) <= 1)
529  return false;
530 
531  if (check_quotes && (get_char(p_in, num_bytes) != '"'))
532  return false;
533 
534  char ch = get_char(p_in, num_bytes);
535 
536  if (ch < 'A' || ch > 'z' || (ch > 'Z' && ch < 'a'))
537  return false;
538 
539  *(p_out++) = ch;
540 
541  for (int i = 1; i < NAME_LENGTH; i++) {
542  if (!check_quotes && (*p_in == ',' || *p_in == ']'))
543  break;
544 
545  ch = get_char(p_in, num_bytes);
546 
547  if (check_quotes && ch == '"')
548  break;
549 
550  if (ch < '0' || ch > 'z' || (ch > '9' && ch < 'A') || (ch > 'Z' && ch < '_') || ch == 0x60)
551  return false;
552 
553  *(p_out++) = ch;
554  }
555 
556  *p_out = 0;
557 
558  if (check_quotes && (ch != '"') && (get_char(p_in, num_bytes) != '"'))
559  return false;
560 
561  if (check_colon) {
562  if (skip_space(p_in, num_bytes) <= 1)
563  return false;
564 
565  if (get_char(p_in, num_bytes) != ':')
566  return false;
567  }
568 
569  return true;
570  }
