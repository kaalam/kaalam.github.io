417  inline StatusCode unwrap_received(pTransaction &p_txn) {
418 
419  pBlock p_blk = (pBlock) &p_txn->p_block->tensor.cell_byte[0];
420  int size = p_txn->p_block->size;
421 
422  if (p_blk->total_bytes == size && p_blk->check_hash()) {
423  pBlock p_new = block_malloc(size);
424  if (p_new == nullptr) {
425  destroy_transaction(p_txn);
426 
427  return SERVICE_ERROR_NO_MEM;
428  }
429  memcpy(p_new, p_blk, size);
430 
431  alloc_bytes -= p_txn->p_block->total_bytes;
432  free(p_txn->p_block);
433 
434  p_txn->p_block = p_new;
435 
436  return SERVICE_NO_ERROR;
437  }
438  if (strnlen((pChar) p_blk, size) != size)
439  return SERVICE_NO_ERROR;
440 
441  pTransaction p_aux;
442 
443  reinterpret_cast <pChar>(p_blk)[size] = 0;
444 
445  if (new_block(p_aux, CELL_TYPE_STRING, nullptr, FILL_WITH_TEXTFILE, 0, (pChar) p_blk, 0) != SERVICE_NO_ERROR) {
446  destroy_transaction(p_txn);
447 
448  return SERVICE_ERROR_NO_MEM;
449  }
450  std::swap(p_txn->p_block, p_aux->p_block);
451 
452  destroy_transaction(p_aux);
453 
454  return SERVICE_NO_ERROR;
455  }

468  inline StatusCode unwrap_received(pTransaction &p_txn, pBlock p_maybe_block, int rec_size) {
469 
470  if (rec_size > sizeof(BlockHeader) && p_maybe_block->total_bytes == rec_size && p_maybe_block->check_hash()) {

642  inline bool get_item_name(pChar &p_in, int &num_bytes, pChar p_out, bool check_quotes = true, bool check_colon = true) {
643  if (skip_space(p_in, num_bytes) <= 1)
644  return false;
645 
646  if (check_quotes && (get_char(p_in, num_bytes) != '"'))
647  return false;
648 
649  char ch = get_char(p_in, num_bytes);
650 
651  if (ch < 'A' || ch > 'z' || (ch > 'Z' && ch < 'a'))
652  return false;
653 
654  *(p_out++) = ch;
655 
656  for (int i = 1; i < NAME_LENGTH; i++) {
657  if (!check_quotes && (*p_in == ',' || *p_in == ']'))
658  break;
659 
660  ch = get_char(p_in, num_bytes);
661 
662  if (check_quotes && ch == '"')
663  break;
664 
665  if (ch < '0' || ch > 'z' || (ch > '9' && ch < 'A') || (ch > 'Z' && ch < '_') || ch == 0x60)
666  return false;
667 
668  *(p_out++) = ch;
669  }
670 
671  *p_out = 0;
672 
673  if (check_quotes && (ch != '"') && (get_char(p_in, num_bytes) != '"'))
674  return false;
675 
676  if (check_colon) {
677  if (skip_space(p_in, num_bytes) <= 1)
678  return false;
679 
680  if (get_char(p_in, num_bytes) != ':')
681  return false;
682  }
683 
684  return true;
685  }
