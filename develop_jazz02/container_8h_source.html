298  inline void lock_container () {
299  int retry = 0;
300  int32_t lock = 0;
301  int32_t next = 1;
302 
303  while (true) {
304  if (_lock_.compare_exchange_weak(lock, next))
305  return;
306 
307  if (++retry > LOCK_NUM_RETRIES_BEFORE_YIELD) {
308  std::this_thread::yield();
309  retry = 0;
310  }
311  }

322  inline StatusCode new_transaction (pTransaction &p_txn) {
323  if (alloc_bytes > warn_alloc_bytes & !alloc_warning_issued) {
324  log_printf(LOG_WARN, "Service Container exceeded RAM %0.2f Mb of %0.2f Mb",
325  (double) alloc_bytes/ONE_MB, (double) warn_alloc_bytes/ONE_MB);

329  lock_container();

363  inline void destroy_internal (pTransaction &p_txn) {
364  if (p_txn->p_block != nullptr) {
365  switch (p_txn->p_block->cell_type) {
366  case CELL_TYPE_INDEX_II:

369  break;
370 
371  case CELL_TYPE_INDEX_IS:
372  p_txn->p_hea->index.index_is.~map();
373 
374  break;
375 
376  case CELL_TYPE_INDEX_SI:
377  p_txn->p_hea->index.index_si.~map();
378 
379  break;
380 
381  case CELL_TYPE_INDEX_SS:
382  p_txn->p_hea->index.index_ss.~map();
383  };
384 
385  alloc_bytes -= p_txn->p_block->total_bytes;
386  free(p_txn->p_block);
387  p_txn->p_block = nullptr;
388  }
389 
390  lock_container();
391 
392  if (pStoredTransaction(p_txn)->p_prev == nullptr)
393  p_alloc = pStoredTransaction(p_txn)->p_next;
394  else
395  pStoredTransaction(p_txn)->p_prev->p_next = pStoredTransaction(p_txn)->p_next;
396 
397  if (pStoredTransaction(p_txn)->p_next != nullptr)
398  pStoredTransaction(p_txn)->p_next->p_prev = pStoredTransaction(p_txn)->p_prev;
399 
400  pStoredTransaction(p_txn)->p_next = p_free;
401 
402  p_free = pStoredTransaction(p_txn);
403  p_txn = nullptr;
404 
405  unlock_container();
