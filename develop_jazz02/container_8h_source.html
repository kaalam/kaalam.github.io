529  inline bool get_item_name(pChar &p_in, int &num_bytes, pChar p_out, bool check_quotes = true, bool check_colon = true) {
530  if (skip_space(p_in, num_bytes) <= 1)
531  return false;
532 
533  if (check_quotes && (get_char(p_in, num_bytes) != '"'))
534  return false;
535 
536  char ch = get_char(p_in, num_bytes);
537 
538  if (ch < 'A' || ch > 'z' || (ch > 'Z' && ch < 'a'))
539  return false;
540 
541  *(p_out++) = ch;
542 
543  for (int i = 1; i < NAME_LENGTH; i++) {
544  if (!check_quotes && (*p_in == ',' || *p_in == ']'))
545  break;
546 
547  ch = get_char(p_in, num_bytes);
548 
549  if (check_quotes && ch == '"')
550  break;
551 
552  if (ch < '0' || ch > 'z' || (ch > '9' && ch < 'A') || (ch > 'Z' && ch < '_') || ch == 0x60)
553  return false;
554 
555  *(p_out++) = ch;
556  }
557 
558  *p_out = 0;
559 
560  if (check_quotes && (ch != '"') && (get_char(p_in, num_bytes) != '"'))
561  return false;
562 
563  if (check_colon) {
564  if (skip_space(p_in, num_bytes) <= 1)
565  return false;
566 
567  if (get_char(p_in, num_bytes) != ':')
568  return false;
569  }
570 
571  return true;
572  }
