306  int retry = 0;
307  int32_t lock = 0;
308  int32_t next = 1;

329  inline StatusCode new_transaction (pTransaction &p_txn) {
330  if (alloc_bytes > warn_alloc_bytes & !alloc_warning_issued) {
331  log_printf(LOG_WARN, "Service Container exceeded RAM %0.2f Mb of %0.2f Mb",
332  (double) alloc_bytes/ONE_MB, (double) warn_alloc_bytes/ONE_MB);
333  alloc_warning_issued = true;

336  lock_container();

346 
347  p_txn->p_block = nullptr;
348  p_txn->p_route = nullptr;

354  pStoredTransaction(p_txn)->p_prev = nullptr;
355

357  p_alloc->p_prev = pStoredTransaction(p_txn);

473  inline bool get_item_name(pChar &p_in, int &num_bytes, pChar p_out) {
474  if (skip_space(p_in, num_bytes) <= 0)
475  return false;
476 
477  if (get_char(p_in, num_bytes) != '"')
478  return false;
479 
480  char ch = get_char(p_in, num_bytes);
481 
482  if (ch < 'A' || ch > 'z' || (ch > 'Z' && ch < 'a'))
483  return false;
484 
485  (p_out++)[0] = ch;
486 
487  for (int i = 1; i < NAME_SIZE; i++) {
488  ch = get_char(p_in, num_bytes);
489 
490  if (ch == '"')
491  break;
492 
493  if (ch < '0' || ch > 'z' || (ch > '9' && ch < 'A') || (ch > 'Z' && ch < '_') || ch == 0x60)
494  return false;
495 
496  (p_out++)[0] = ch;
497  }
498 
499  if (ch != '"' && (get_char(p_in, num_bytes) != '"'))
500  return false;
501 
502  if (skip_space(p_in, num_bytes) <= 0)
503  return false;
504 
505  if (get_char(p_in, num_bytes) != ':')
506  return false;
507 
508  return true;
509  }
