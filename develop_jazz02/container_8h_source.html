229  inline void lock_container () {
230  int retry = 0;
231  int32_t lock = 0;
232  int32_t next = 1;
233 
234  while (true) {
235  if (_lock_.compare_exchange_weak(lock, next))
236  return;
237 
238  if (++retry > LOCK_NUM_RETRIES_BEFORE_YIELD) {
239  std::this_thread::yield();
240  retry = 0;
241  }
242  }
243  }

253  inline StatusCode new_transaction (pTransaction &p_keeper) {
254  if (alloc_bytes > warn_alloc_bytes & !alloc_warning_issued) {
255  log_printf(LOG_WARN, "Service Container exceeded RAM %0.2f Mb of %0.2f Mb",
256  (double) alloc_bytes/ONE_MB, (double) warn_alloc_bytes/ONE_MB);
257  alloc_warning_issued = true;
258  }
259 
260  lock_container();
261 
262  if (p_free == nullptr) {
263  unlock_container();
264  p_keeper = nullptr;
265  return SERVICE_ERROR_NO_MEM;
266  }
267 
268  p_keeper = p_free;
269  p_free = p_free->p_next;
270 
271  p_keeper->p_block = nullptr;
272  p_keeper->status = BLOCK_STATUS_EMPTY;
273  p_keeper->_lock_ = 0;
274 
275  pStoredTransaction(p_keeper)->p_next = p_alloc;
276  pStoredTransaction(p_keeper)->p_prev = nullptr;
277 
278  if (p_alloc != nullptr)
279  p_alloc->p_prev = pStoredTransaction(p_keeper);
280 
281  p_alloc = pStoredTransaction(p_keeper);
282 
283  unlock_container();
284  return SERVICE_NO_ERROR;
285  }

289  inline void destroy_transaction (pTransaction &p_keeper) {
290  if (p_keeper->p_block != nullptr) {
291  alloc_bytes -= p_keeper->p_block->total_bytes;
292  free(p_keeper->p_block);
293  p_keeper->p_block = nullptr;
294  }
295 
296  lock_container();
297 
298  if (pStoredTransaction(p_keeper)->p_prev == nullptr)
299  p_alloc = pStoredTransaction(p_keeper)->p_next;
300  else
301  pStoredTransaction(p_keeper)->p_prev->p_next = pStoredTransaction(p_keeper)->p_next;
302 
303  if (pStoredTransaction(p_keeper)->p_next != nullptr)
304  pStoredTransaction(p_keeper)->p_next->p_prev = pStoredTransaction(p_keeper)->p_prev;
305 
306  pStoredTransaction(p_keeper)->p_next = p_free;
307 
308  p_free = pStoredTransaction(p_keeper);
309  p_keeper = nullptr;
310 
311  unlock_container();
312  }
