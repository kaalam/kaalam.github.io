432 inline StatusCode unwrap_received(pTransaction &p_txn) {
433
434 if (p_txn->p_block->cell_type == CELL_TYPE_STRING)
435 if (p_txn->p_block->size == 1)
436 return SERVICE_NO_ERROR; // The block is already a string, we just leave it as it is.
437 else {
438 destroy_transaction(p_txn);
439
440 return SERVICE_ERROR_WRONG_TYPE;
441 }
442
443 if ((p_txn->p_block->cell_type != CELL_TYPE_BYTE) || (p_txn->p_block->rank != 1)) {
444 destroy_transaction(p_txn);
445
446 return SERVICE_ERROR_WRONG_TYPE;
447 }
448
449 pBlock p_blk = (pBlock) &p_txn->p_block->tensor.cell_byte[0];
450 int size = p_txn->p_block->size;
451
452 if (p_blk->total_bytes == size && p_blk->check_hash()) {
453 pBlock p_new = block_malloc(size);
454 if (p_new == nullptr) {
455 destroy_transaction(p_txn);
456
457 return SERVICE_ERROR_NO_MEM;
458 }
459 memcpy(p_new, p_blk, size);
460
461 alloc_bytes -= p_txn->p_block->total_bytes;
462 free(p_txn->p_block);
463
464 p_txn->p_block = p_new;
465
466 return SERVICE_NO_ERROR;
467 }
468
469 int block_len = strnlen((pChar) p_blk, size);
470 if (block_len < size - 1)
471 return SERVICE_NO_ERROR; // This is when the content is not a string, we just leave it as it is.
472
473 pTransaction p_aux;
474
475 reinterpret_cast <pChar>(p_blk)[size] = 0;
476
477 if (new_block(p_aux, CELL_TYPE_STRING, nullptr, FILL_WITH_TEXTFILE, 0, (pChar) p_blk, 0) != SERVICE_NO_ERROR) {
478 destroy_transaction(p_txn);
479
480 return SERVICE_ERROR_NO_MEM;
481 }
482 std::swap(p_txn->p_block, p_aux->p_block);
483
484 destroy_transaction(p_aux);
485
486 return SERVICE_NO_ERROR;
487 }

500 inline StatusCode unwrap_received(pTransaction &p_txn, pBlock p_maybe_block, int rec_size) {
501
502 if (rec_size > sizeof(BlockHeader) && p_maybe_block->total_bytes == rec_size && p_maybe_block->check_hash()) {
503 int ret = new_transaction(p_txn);
504
505 if (ret != SERVICE_NO_ERROR)
506 return ret;
507
508 pBlock p_new = block_malloc(rec_size);
509 if (p_new == nullptr) {
510 destroy_transaction(p_txn);
511
512 return SERVICE_ERROR_NO_MEM;
513 }
514 memcpy(p_new, p_maybe_block, rec_size);
515
516 p_txn->p_block = p_new;
517 p_txn->status = BLOCK_STATUS_READY;
518
519 return SERVICE_NO_ERROR;
520 }
521 if (strnlen((pChar) p_maybe_block, rec_size) != rec_size) {
522 int dim[MAX_TENSOR_RANK] = {rec_size, 0};
523 int ret = new_block(p_txn, CELL_TYPE_BYTE, dim, FILL_NEW_DONT_FILL);
524
525 if (ret == SERVICE_NO_ERROR)
526 memcpy(&p_txn->p_block->tensor.cell_byte[0], p_maybe_block, rec_size);
527
528 return ret;
529 }
530
531 return new_block(p_txn, CELL_TYPE_STRING, nullptr, FILL_WITH_TEXTFILE, 0, (pChar) p_maybe_block, 0);
532 }

581 inline void lock_container(int total_times = 0) {
582 int retry = 0;
583
584 while (true) {
585 int32_t lock = 0;
586 int32_t next = 1;
587 if (_lock_.compare_exchange_weak(lock, next))
588 return;
589
590 if (++retry > LOCK_NUM_RETRIES_BEFORE_YIELD) {
591 std::this_thread::yield();
592 retry = 0;
593 if (total_times) {
594 if (--total_times == 0)
595 return;
596 }
597 }
598 }

643 inline int skip_space(pChar &p_in, int &num_bytes) {
644 while (num_bytes > 0) {
645 if (*p_in == 0)
646 return 0;
647
648 if (*p_in == ' ' || *p_in == '\t' || *p_in == '\n') {
649 p_in++;
650 num_bytes--;
651 } else
652 break;
653 }
654
655 return num_bytes;
656 }

666 if (num_bytes < 1)
667 return 0;
668
669 num_bytes--;
670
671 return *(p_in++);
672 }

685 inline bool get_item_name(pChar &p_in,
686 int &num_bytes,
687 pChar p_out,
688 bool check_quotes = true,
689 bool check_colon = true,
690 bool dot_ended = false) {
691 if (skip_space(p_in, num_bytes) <= 1)
692 return false;
693
694 if (check_quotes && (get_char(p_in, num_bytes) != '"'))
695 return false;
696
697 char ch = get_char(p_in, num_bytes);
698
699 if (ch < 'A' || ch > 'z' || (ch > 'Z' && ch < 'a'))
700 return false;
701
702 *(p_out++) = ch;
703
704 for (int i = 1; i < NAME_LENGTH; i++) {
705 if (!check_quotes && (*p_in == ',' || *p_in == ']'))
706 break;
707
708 if (dot_ended && (*p_in == '.'))
709 break;
710
711 ch = get_char(p_in, num_bytes);
712
713 if ((check_quotes || dot_ended) && ch == '"')
714 break;
715
716 if (ch < '0' || ch > 'z' || (ch > '9' && ch < 'A') || (ch > 'Z' && ch < '_') || ch == 0x60)
717 return false;
718
719 *(p_out++) = ch;
720 }
721
722 *p_out = 0;
723
724 if (check_quotes && (ch != '"') && (get_char(p_in, num_bytes) != '"'))
725 return false;
726
727 if (check_colon) {
728 if (skip_space(p_in, num_bytes) <= 1)
729 return false;
730
731 if (get_char(p_in, num_bytes) != ':')
732 return false;
733 }
734
735 return true;
736 }

977 *(p_ret++) = '[';

986 } else

988

991 *(p_ret++) = ' ';

993 }

995 *(p_ret++) = ']';

997 *p_ret = 0;

1021 inline void separator(int rank_1, int dim[], int idx[], pChar &p_ret) {
1022 for (int i = rank_1; i >= 0; i--) {
1023 idx[i]++;
1024
1025 if (idx[i] == dim[i]) {
1026 idx[i] = 0;
1027 *(p_ret++) = ']';
1028 } else {
1029 *(p_ret++) = ',';
1030 *(p_ret++) = ' ';
1031
1032 for (int j = 0; j < rank_1 - i; j++)
1033 *(p_ret++) = '[';
1034
1035 break;
