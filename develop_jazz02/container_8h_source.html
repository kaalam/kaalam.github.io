426 inline StatusCode unwrap_received(pTransaction &p_txn) {
427
428 pBlock p_blk = (pBlock) &p_txn->p_block->tensor.cell_byte[0];
429 int size = p_txn->p_block->size;
430
431 if (p_blk->total_bytes == size && p_blk->check_hash()) {
432 pBlock p_new = block_malloc(size);
433 if (p_new == nullptr) {
434 destroy_transaction(p_txn);
435
436 return SERVICE_ERROR_NO_MEM;
437 }
438 memcpy(p_new, p_blk, size);
439
440 alloc_bytes -= p_txn->p_block->total_bytes;
441 free(p_txn->p_block);
442
443 p_txn->p_block = p_new;
444
445 return SERVICE_NO_ERROR;
446 }
447 if (strnlen((pChar) p_blk, size) != size)
448 return SERVICE_NO_ERROR;
449
450 pTransaction p_aux;
451
452 reinterpret_cast <pChar>(p_blk)[size] = 0;
453
454 if (new_block(p_aux, CELL_TYPE_STRING, nullptr, FILL_WITH_TEXTFILE, 0, (pChar) p_blk, 0) != SERVICE_NO_ERROR) {
455 destroy_transaction(p_txn);
456
457 return SERVICE_ERROR_NO_MEM;
458 }
459 std::swap(p_txn->p_block, p_aux->p_block);
460
461 destroy_transaction(p_aux);
462
463 return SERVICE_NO_ERROR;
464 }

477 inline StatusCode unwrap_received(pTransaction &p_txn, pBlock p_maybe_block, int rec_size) {
478
479 if (rec_size > sizeof(BlockHeader) && p_maybe_block->total_bytes == rec_size && p_maybe_block->check_hash()) {
480 int ret = new_transaction(p_txn);
481
482 if (ret != SERVICE_NO_ERROR)
483 return ret;
484
485 pBlock p_new = block_malloc(rec_size);
486 if (p_new == nullptr) {
487 destroy_transaction(p_txn);
488
489 return SERVICE_ERROR_NO_MEM;
490 }
491 memcpy(p_new, p_maybe_block, rec_size);
492
493 p_txn->p_block = p_new;
494 p_txn->status = BLOCK_STATUS_READY;
495
496 return SERVICE_NO_ERROR;
497 }
498 if (strnlen((pChar) p_maybe_block, rec_size) != rec_size) {
499 int dim[MAX_TENSOR_RANK] = {rec_size, 0};
500 int ret = new_block(p_txn, CELL_TYPE_BYTE, dim, FILL_NEW_DONT_FILL);
501
502 if (ret == SERVICE_NO_ERROR)
503 memcpy(&p_txn->p_block->tensor.cell_byte[0], p_maybe_block, rec_size);
504
505 return ret;
506 }
507
508 return new_block(p_txn, CELL_TYPE_STRING, nullptr, FILL_WITH_TEXTFILE, 0, (pChar) p_maybe_block, 0);
509 }

555 inline void lock_container() {
556 int retry = 0;
557 int32_t lock = 0;
558 int32_t next = 1;
559
560 while (true) {
561 if (_lock_.compare_exchange_weak(lock, next))
562 return;
563
564 if (++retry > LOCK_NUM_RETRIES_BEFORE_YIELD) {
565 std::this_thread::yield();
566 retry = 0;
567 }
568 }
569 }

610 inline int skip_space(pChar &p_in, int &num_bytes) {
611 while (num_bytes > 0) {
612 if (*p_in == 0)
613 return 0;
614
615 if (*p_in == ' ' || *p_in == '\t' || *p_in == '\n') {
616 p_in++;
617 num_bytes--;
618 } else
619 break;
620 }
621
622 return num_bytes;
623 }

636 num_bytes--;
637
638 return *(p_in++);

651 inline bool get_item_name(pChar &p_in, int &num_bytes, pChar p_out, bool check_quotes = true, bool check_colon = true) {
652 if (skip_space(p_in, num_bytes) <= 1)
653 return false;
654
655 if (check_quotes && (get_char(p_in, num_bytes) != '"'))
656 return false;
657
658 char ch = get_char(p_in, num_bytes);
659
660 if (ch < 'A' || ch > 'z' || (ch > 'Z' && ch < 'a'))
661 return false;
662
663 *(p_out++) = ch;
664
665 for (int i = 1; i < NAME_LENGTH; i++) {
666 if (!check_quotes && (*p_in == ',' || *p_in == ']'))
667 break;
668
669 ch = get_char(p_in, num_bytes);
670
671 if (check_quotes && ch == '"')
672 break;
673
674 if (ch < '0' || ch > 'z' || (ch > '9' && ch < 'A') || (ch > 'Z' && ch < '_') || ch == 0x60)
675 return false;
676
677 *(p_out++) = ch;
678 }
679
680 *p_out = 0;
681
682 if (check_quotes && (ch != '"') && (get_char(p_in, num_bytes) != '"'))
683 return false;
684
685 if (check_colon) {
686 if (skip_space(p_in, num_bytes) <= 1)
687 return false;
688
689 if (get_char(p_in, num_bytes) != ':')
690 return false;
691 }
692
693 return true;
694 }

978 inline void separator(int rank_1, int dim[], int idx[], pChar &p_ret) {
979 for (int i = rank_1; i >= 0; i--) {
980 idx[i]++;
981
982 if (idx[i] == dim[i]) {
983 idx[i] = 0;
984 *(p_ret++) = ']';
985 } else {
986 *(p_ret++) = ',';
987 *(p_ret++) = ' ';
988
989 for (int j = 0; j < rank_1 - i; j++)
990 *(p_ret++) = '[';
991
992 break;
993 }
994 }
995 *p_ret = 0;
996 }

1006 inline int separator_len(int rank_1, int dim[], int idx[]) {
1007 int ret = 0;
1008
1009 for (int i = rank_1; i >= 0; i--) {
1010 idx[i]++;
1011
1012 if (idx[i] == dim[i]) {
1013 idx[i] = 0;
1014 ret++;
1015 } else {
1016 ret += 2;
1017
1018 if (i == rank_1)
1019 return ret;
1020
1021 ret += rank_1 - i;
1022
1023 return ret;
1024 }
1025 }
