348  inline void lock_container () {
349  int retry = 0;
350  int32_t lock = 0;
351  int32_t next = 1;
352 
353  while (true) {
354  if (_lock_.compare_exchange_weak(lock, next))
355  return;
356 
357  if (++retry > LOCK_NUM_RETRIES_BEFORE_YIELD) {
358  std::this_thread::yield();
359  retry = 0;
360  }
361  }
362  }

366  void unlock_container () {
367  _lock_ = 0;
368  }

372  inline StatusCode new_keeper (pBlockKeeper &p_keeper) {
373  if (alloc_bytes > warn_alloc_bytes & !alloc_warning_issued) {
374  log_printf(LOG_WARN, "Service Container exceeded RAM %0.2f Mb of %0.2f Mb",
375  (double) alloc_bytes/ONE_MB, (double) warn_alloc_bytes/ONE_MB);
376  alloc_warning_issued = true;
377  }
378 
379  lock_container();
380 
381  if (p_free == nullptr) {
382  unlock_container();
383  p_keeper = nullptr;
384  return SERVICE_ERROR_NO_MEM;
385  }
386 
387  p_keeper = p_free;
388  p_free = p_free->data.deque.p_next;
389 
390  p_keeper->p_block = nullptr;
391  p_keeper->status = BLOCK_STATUS_EMPTY;
392  p_keeper->_lock_ = 0;
393 
394  p_keeper->data.deque.p_next = p_alloc;
395  p_keeper->data.deque.p_prev = nullptr;
396 
397  if (p_alloc != nullptr)
398  p_alloc->data.deque.p_prev = p_keeper;
399 
400  p_alloc = p_keeper;
401 
402  unlock_container();
403  return SERVICE_NO_ERROR;
404  }
405 
408  inline void destroy_keeper (pBlockKeeper &p_keeper) {
409  if (p_keeper->p_block != nullptr) {
410  alloc_bytes -= p_keeper->p_block->total_bytes;
411  free(p_keeper->p_block);
412  p_keeper->p_block = nullptr;
413  }
414 
415  lock_container();
416 
417  if (p_keeper->data.deque.p_prev == nullptr)
418  p_alloc = p_keeper->data.deque.p_next;
419  else
420  p_keeper->data.deque.p_prev->data.deque.p_next = p_keeper->data.deque.p_next;
421 
422  if (p_keeper->data.deque.p_next != nullptr)
423  p_keeper->data.deque.p_next->data.deque.p_prev = p_keeper->data.deque.p_prev;
424 
425  p_keeper->data.deque.p_next = p_free;
426 
427  p_free = p_keeper;
428  p_keeper = nullptr;
429 
430  unlock_container();
431  }
