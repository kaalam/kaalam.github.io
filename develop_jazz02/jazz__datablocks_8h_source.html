157  inline void set_dimensions(int *pDim) {
158  rank = JAZZ_MAX_TENSOR_RANK;
159  int j = 1;
160  for (int i = JAZZ_MAX_TENSOR_RANK -1; i > 0; i--)
161  if (pDim[i] > 1) { dim_offs[i] = j; j *= pDim[i]; } else { j = 1; dim_offs[i] = 0; rank = i; }
162  if (pDim[0] > 1) { dim_offs[0] = j; j *= pDim[0]; } else { j = pDim[0]; dim_offs[0] = 1; rank = 1; }
163  size = j;
164  }
165 
173  inline void get_dimensions(int *pDim) {
174  int j = size;
175  for (int i = 0; i < JAZZ_MAX_TENSOR_RANK; i++)
176  if (i < rank) { pDim[i] = j/dim_offs[i]; j = dim_offs[i]; } else pDim[i] = 0;
177  }
178 
185  inline bool validate_index(int *pIndex) {
186  int j = size;
187  for (int i = 0; i < rank; i++) {
188  if (pIndex[i] < 0 || pIndex[i]*dim_offs[i] >= j) return false;
189  j = dim_offs[i];
190  }
191  return true;
192  }
193 
200  inline int validate_offset(int offset) { return offset >=0 & offset < size; }

208  inline int get_offset(int *pIndex) {

311  inline void set_attributes(AllAttributes &all_att) {
312  if (num_attributes) return;
313 
314  num_attributes = all_att.size();
315  init_string_buffer();
316 
317  int i = 0;
318  int *ptk = pAttribute_keys();
319  pJazzStringBuffer psb = pStringBuffer();
320  for(AllAttributes::iterator it = all_att.begin(); it != all_att.end(); ++it)
321  {
322  if (i < num_attributes) {
323  ptk[i] = it->first;
324  ptk[i + num_attributes] = get_string_offset(psb, it->second);
325  }
326  i++;

349  inline void init_string_buffer() {
350  pJazzStringBuffer psb = pStringBuffer();
351 
352  int buff_size = total_bytes - ((uintptr_t) psb - (uintptr_t) &cell_type) - sizeof(JazzStringBuffer);
353  if (buff_size < 4) {
354  psb->alloc_failed = true;
355  return;
356  }
357  memset(psb, 0, sizeof(JazzStringBuffer) + 4); // This also clears psb->buffer[0..3]
358  // psb->buffer[0] = 0; // JAZZ_STRING_NA
359  // psb->buffer[1] = 0; // JAZZ_STRING_EMPTY
360  // psb->buffer[2] = 0; // The end of string for searching (will change once a string is inserted)
361  psb->buffer_size = buff_size;
362  psb->last_idx = 2; // Where the first string will be inserted
363  }
