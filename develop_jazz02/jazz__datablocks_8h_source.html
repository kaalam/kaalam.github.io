179  // Methods on indices.
180 
191  inline void set_dimensions(int *pDim) {
192  rank = JAZZ_MAX_TENSOR_RANK;
193  int j = 1;
194  for (int i = JAZZ_MAX_TENSOR_RANK -1; i > 0; i--)
195  if (pDim[i] > 1) { range.dim[i] = j; j *= pDim[i]; } else { j = 1; range.dim[i] = 0; rank = i; }
196  range.dim[0] = j;
197  j *= pDim[0];
198  size = j;
199  }
200 
208  inline void get_dimensions(int *pDim) {
209  int j = size;
210  for (int i = 0; i < JAZZ_MAX_TENSOR_RANK; i++)
211  if (i < rank) { pDim[i] = j/range.dim[i]; j = range.dim[i]; } else pDim[i] = 0;
212  }
213 
220  inline bool validate_index(int *pIndex) {
221  int j = size;
222  for (int i = 0; i < rank; i++) {
223  if (pIndex[i] < 0 || pIndex[i]*range.dim[i] >= j) return false;
224  j = range.dim[i];
225  }
226  return true;
227  }
228 
235  inline int validate_offset(int offset) { return offset >=0 & offset < size; }
236 
243  inline int get_offset(int *pIndex) {
244  int j = 0;
245  for (int i = 0; i < rank; i++) j += pIndex[i]*range.dim[i];
246  return j;
247  }
248 
254  inline void get_index(int offset, int *pIndex) {

346  inline void set_attributes(AllAttributes *all_att) {
347  if (num_attributes) return;
348 
349  num_attributes = all_att->size();
350  init_string_buffer();
351 
352  int i = 0;
353  int *ptk = p_attribute_keys();
354  pJazzStringBuffer psb = p_string_buffer();
355  for (AllAttributes::iterator it = all_att->begin(); it != all_att->end(); ++it) {
356  if (i < num_attributes) {
357  ptk[i] = it->first;
358  ptk[i + num_attributes] = get_string_offset(psb, it->second);
359  }
360  i++;
361  }
362  }

383  inline void init_string_buffer() {
384  pJazzStringBuffer psb = p_string_buffer();
385 
386  int buff_size = total_bytes - ((uintptr_t) psb - (uintptr_t) &cell_type) - sizeof(JazzStringBuffer);
387  if (buff_size < 4) {
388  psb->alloc_failed = true;
389  return;
390  }
391  memset(psb, 0, sizeof(JazzStringBuffer) + 4); // This also clears psb->buffer[0..3]
392  // psb->buffer[0] = 0; // JAZZ_STRING_NA
393  // psb->buffer[1] = 0; // JAZZ_STRING_EMPTY
394  // psb->buffer[2] = 0; // The end of string for searching (will change once a string is inserted)
395  psb->buffer_size = buff_size;
396  psb->last_idx = 2; // Where the first string will be inserted
397  }
