147  inline void set_dimensions(int *pDim) {
148  rank = JAZZ_MAX_TENSOR_RANK;
149  int j = 1;
150  for (int i = JAZZ_MAX_TENSOR_RANK -1; i > 0; i--)
151  if (pDim[i] > 1) { dim_offs[i] = j; j *= pDim[i]; } else { j = 1; dim_offs[i] = 0; rank = i; }
152  if (pDim[0] > 1) { dim_offs[0] = j; j *= pDim[0]; } else { j = pDim[0]; dim_offs[0] = 1; rank = 1; }
153  size = j;
154  }
155 
163  inline void get_dimensions(int *pDim) {
164  int j = size;
165  for (int i = 0; i < JAZZ_MAX_TENSOR_RANK; i++)
166  if (i < rank) { pDim[i] = j/dim_offs[i]; j = dim_offs[i]; } else pDim[i] = 0;
167  }

175  inline bool validate_index(int *pIndex) {
176  int j = size;
177  for (int i = 0; i < rank; i++) {
178  if (pIndex[i] < 0 || pIndex[i]*dim_offs[i] >= j) return false;
179  j = dim_offs[i];
180  }
181  return true;
182  }

339  inline void init_string_buffer() {
340  pJazzStringBuffer psb = pStringBuffer();
341 
342  int buff_size = total_bytes - ((uintptr_t) psb - (uintptr_t) &cell_type) - sizeof(JazzStringBuffer);
343  if (buff_size < 4) {
344  psb->alloc_failed = true;
345  return;
346  }
347  memset(psb, 0, sizeof(JazzStringBuffer) + 4); // This also clears psb->buffer[0..3]
348  // psb->buffer[0] = 0; // JAZZ_STRING_NA
349  // psb->buffer[1] = 0; // JAZZ_STRING_EMPTY
350  // psb->buffer[2] = 0; // The end of string for searching (will change once a string is inserted)
351  psb->buffer_size = buff_size;
352  psb->last_idx = 2; // Where the first string will be inserted
353  }
