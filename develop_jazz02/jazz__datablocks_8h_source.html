178  // Methods on indices.
179 
190  inline void set_dimensions(int *pDim) {
191  rank = JAZZ_MAX_TENSOR_RANK;
192  int j = 1;
193  for (int i = JAZZ_MAX_TENSOR_RANK -1; i > 0; i--)
194  if (pDim[i] > 1) { range.dim[i] = j; j *= pDim[i]; } else { j = 1; range.dim[i] = 0; rank = i; }
195  range.dim[0] = j;
196  j *= pDim[0];
197  size = j;
198  }
199 
207  inline void get_dimensions(int *pDim) {
208  int j = size;
209  for (int i = 0; i < JAZZ_MAX_TENSOR_RANK; i++)
210  if (i < rank) { pDim[i] = j/range.dim[i]; j = range.dim[i]; } else pDim[i] = 0;
211  }
212 
219  inline bool validate_index(int *pIndex) {
220  int j = size;
221  for (int i = 0; i < rank; i++) {
222  if (pIndex[i] < 0 || pIndex[i]*range.dim[i] >= j) return false;
223  j = range.dim[i];
224  }
225  return true;
226  }

345  inline void set_attributes(AllAttributes *all_att) {
346  if (num_attributes) return;
347 
348  num_attributes = all_att->size();
349  init_string_buffer();
350 
351  int i = 0;
352  int *ptk = p_attribute_keys();
353  pJazzStringBuffer psb = p_string_buffer();
354  for (AllAttributes::iterator it = all_att->begin(); it != all_att->end(); ++it) {
355  if (i < num_attributes) {
356  ptk[i] = it->first;
357  ptk[i + num_attributes] = get_string_offset(psb, it->second);
358  }
359  i++;
360  }
361  }

374  (*all_att)[ptk[i]] = reinterpret_cast<char *>(&psb->buffer[ptk[i + num_attributes]]);

382  inline void init_string_buffer() {
383  pJazzStringBuffer psb = p_string_buffer();
384 
385  int buff_size = total_bytes - ((uintptr_t) psb - (uintptr_t) &cell_type) - sizeof(JazzStringBuffer);
386  if (buff_size < 4) {
387  psb->alloc_failed = true;
388  return;
389  }
390  memset(psb, 0, sizeof(JazzStringBuffer) + 4); // This also clears psb->buffer[0..3]
391  // psb->buffer[0] = 0; // JAZZ_STRING_NA
392  // psb->buffer[1] = 0; // JAZZ_STRING_EMPTY
393  // psb->buffer[2] = 0; // The end of string for searching (will change once a string is inserted)
394  psb->buffer_size = buff_size;
395  psb->last_idx = 2; // Where the first string will be inserted
396  }
