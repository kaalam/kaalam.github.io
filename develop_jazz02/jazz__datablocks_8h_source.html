164  inline void set_dimensions(int *pDim) {
165  rank = JAZZ_MAX_TENSOR_RANK;
166  int j = 1;
167  for (int i = JAZZ_MAX_TENSOR_RANK -1; i > 0; i--)
168  if (pDim[i] > 1) { dim_offs[i] = j; j *= pDim[i]; } else { j = 1; dim_offs[i] = 0; rank = i; }
169  if (pDim[0] > 1) { dim_offs[0] = j; j *= pDim[0]; } else { j = pDim[0]; dim_offs[0] = 1; rank = 1; }
170  size = j;
171  }
172 
180  inline void get_dimensions(int *pDim) {
181  int j = size;
182  for (int i = 0; i < JAZZ_MAX_TENSOR_RANK; i++)
183  if (i < rank) { pDim[i] = j/dim_offs[i]; j = dim_offs[i]; } else pDim[i] = 0;
184  }
185 
192  inline bool validate_index(int *pIndex) {
193  int j = size;
194  for (int i = 0; i < rank; i++) {
195  if (pIndex[i] < 0 || pIndex[i]*dim_offs[i] >= j) return false;
196  j = dim_offs[i];
197  }
198  return true;
199  }
200 
207  inline int validate_offset(int offset) { return offset >=0 & offset < size; }
208 
215  inline int get_offset(int *pIndex) {
216  int j = 0;
217  for (int i = 0; i < rank; i++) j += pIndex[i]*dim_offs[i];

226  inline void get_index(int offset, int *pIndex) {

318  inline void set_attributes(AllAttributes &all_att) {
319  if (num_attributes) return;

322  init_string_buffer();
323

356  inline void init_string_buffer() {
357  pJazzStringBuffer psb = pStringBuffer();
358 
359  int buff_size = total_bytes - ((uintptr_t) psb - (uintptr_t) &cell_type) - sizeof(JazzStringBuffer);
360  if (buff_size < 4) {
361  psb->alloc_failed = true;
362  return;
363  }
364  memset(psb, 0, sizeof(JazzStringBuffer) + 4); // This also clears psb->buffer[0..3]
365  // psb->buffer[0] = 0; // JAZZ_STRING_NA
366  // psb->buffer[1] = 0; // JAZZ_STRING_EMPTY
367  // psb->buffer[2] = 0; // The end of string for searching (will change once a string is inserted)
368  psb->buffer_size = buff_size;
369  psb->last_idx = 2; // Where the first string will be inserted
370  }
