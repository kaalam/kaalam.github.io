179  inline void set_dimensions(int *pDim) {
180  rank = JAZZ_MAX_TENSOR_RANK;
181  int j = 1;
182  for (int i = JAZZ_MAX_TENSOR_RANK -1; i > 0; i--)
183  if (pDim[i] > 1) { dim_offs[i] = j; j *= pDim[i]; } else { j = 1; dim_offs[i] = 0; rank = i; }
184  dim_offs[0] = j;
185  j *= pDim[0];
186  size = j;
187  }
188 
196  inline void get_dimensions(int *pDim) {
197  int j = size;
198  for (int i = 0; i < JAZZ_MAX_TENSOR_RANK; i++)
199  if (i < rank) { pDim[i] = j/dim_offs[i]; j = dim_offs[i]; } else pDim[i] = 0;
200  }
201 
208  inline bool validate_index(int *pIndex) {
209  int j = size;
210  for (int i = 0; i < rank; i++) {
211  if (pIndex[i] < 0 || pIndex[i]*dim_offs[i] >= j) return false;
212  j = dim_offs[i];
213  }
214  return true;
215  }
216 
223  inline int validate_offset(int offset) { return offset >=0 & offset < size; }
224 
231  inline int get_offset(int *pIndex) {
232  int j = 0;
233  for (int i = 0; i < rank; i++) j += pIndex[i]*dim_offs[i];
234  return j;
235  }
236 
242  inline void get_index(int offset, int *pIndex) {
243  for (int i = 0; i < rank; i++) { pIndex[i] = offset/dim_offs[i]; offset -= pIndex[i]*dim_offs[i]; }
244  }

334  inline void set_attributes(AllAttributes &all_att) {
335  if (num_attributes) return;
336 
337  num_attributes = all_att.size();
338  init_string_buffer();
339 
340  int i = 0;
341  int *ptk = pAttribute_keys();
342  pJazzStringBuffer psb = pStringBuffer();
343  for(AllAttributes::iterator it = all_att.begin(); it != all_att.end(); ++it)
344  {
345  if (i < num_attributes) {
346  ptk[i] = it->first;
347  ptk[i + num_attributes] = get_string_offset(psb, it->second);
348  }
349  i++;
350  }
351  }

372  inline void init_string_buffer() {
373  pJazzStringBuffer psb = pStringBuffer();
374 
375  int buff_size = total_bytes - ((uintptr_t) psb - (uintptr_t) &cell_type) - sizeof(JazzStringBuffer);
376  if (buff_size < 4) {
377  psb->alloc_failed = true;
378  return;
379  }
380  memset(psb, 0, sizeof(JazzStringBuffer) + 4); // This also clears psb->buffer[0..3]
381  // psb->buffer[0] = 0; // JAZZ_STRING_NA
382  // psb->buffer[1] = 0; // JAZZ_STRING_EMPTY
383  // psb->buffer[2] = 0; // The end of string for searching (will change once a string is inserted)
384  psb->buffer_size = buff_size;
385  psb->last_idx = 2; // Where the first string will be inserted
386  }
