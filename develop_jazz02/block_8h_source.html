120  inline void set_dimensions(int *pDim) {
121  rank = MAX_TENSOR_RANK;
122  int j = 1;
123  for (int i = MAX_TENSOR_RANK -1; i > 0; i--)
124  if (pDim[i] > 1) { range.dim[i] = j; j *= pDim[i]; } else { j = 1; range.dim[i] = 0; rank = i; }
125  range.dim[0] = j;
126  j *= pDim[0];
127  size = j;
128  }
