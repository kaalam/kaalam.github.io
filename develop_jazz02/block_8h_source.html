115  inline void set_dimensions(int *p_dim) {
116  rank = MAX_TENSOR_RANK;
117  int j = 1;
118  for (int i = MAX_TENSOR_RANK - 1; i > 0; i--)
119  if (p_dim[i] > 0) { range.dim[i] = j; j *= p_dim[i]; } else { j = 1; range.dim[i] = 0; rank = i; }
120  range.dim[0] = j;
121  j *= p_dim[0];
122  size = j;
123  }
