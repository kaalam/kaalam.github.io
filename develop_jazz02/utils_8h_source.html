185inline bool valid_name(pChar p_name) {
186 char ch = p_name++[0];
187
188 if (!ch)
189 return false;
190
191 if (ch < 'A' || ch > 'z' || (ch > 'Z' && ch < 'a'))
192 return false;
193
194 for (int i = 1; i < NAME_SIZE; i++) {
195 char ch = p_name++[0];
196
197 if (!ch)
198 return true;
199
200 if (ch < '0' || ch > 'z' || (ch > '9' && ch < 'A') || (ch > 'Z' && ch < '_') || ch == 0x60)
201 return false;
202 }
203 return false;
204}
