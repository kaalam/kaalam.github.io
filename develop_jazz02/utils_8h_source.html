161 inline bool valid_name(pChar p_name) {
162  char ch = p_name++[0];
163 
164  if (!ch)
165  return false;
166 
167  if (ch < 'A' || ch > 'z' || (ch > 'Z' && ch < 'a'))
168  return false;
169 
170  for (int i = 1; i < NAME_SIZE; i++) {
171  char ch = p_name++[0];
172 
173  if (!ch)
174  return true;
175 
176  if (ch < '0' || ch > 'z' || (ch > '9' && ch < 'A') || (ch > 'Z' && ch < '_') || ch == 0x60)
177  return false;
178  }
179  return false;
180 }
