188inline bool valid_name(pChar p_name) {
189 char ch = p_name++[0];
190
191 if (!ch)
192 return false;
193
194 if (ch < 'A' || ch > 'z' || (ch > 'Z' && ch < 'a'))
195 return false;
196
197 for (int i = 1; i < NAME_SIZE; i++) {
198 char ch = p_name++[0];
199
200 if (!ch)
201 return true;
202
203 if (ch < '0' || ch > 'z' || (ch > '9' && ch < 'A') || (ch > 'Z' && ch < '_') || ch == 0x60)
204 return false;
205 }
206 return false;
207}
