293  inline StatusCode curl_get(pTransaction &p_txn, void *url, Index *p_idx = nullptr) {
294  CURL *curl;
295  CURLcode c_ret;
296 
297  curl = curl_easy_init();
298  if (!curl)
299  return SERVICE_ERROR_NOT_READY;
300 
301  GetBuffer buff = {};
302 
303  curl_easy_setopt(curl, CURLOPT_URL, url);
304  curl_easy_setopt(curl, CURLOPT_VERBOSE, 0);
305  curl_easy_setopt(curl, CURLOPT_NOPROGRESS, 1);
306  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, get_callback);
307  curl_easy_setopt(curl, CURLOPT_WRITEDATA, (void *) &buff);
308 
309  if (p_idx != nullptr) {
310  Index:: iterator it;
311  if ((it = p_idx->find("CURLOPT_USERNAME")) != p_idx->end())
312  curl_easy_setopt(curl, CURLOPT_USERNAME, it->second.c_str());
313 
314  if ((it = p_idx->find("CURLOPT_USERPWD")) != p_idx->end())
315  curl_easy_setopt(curl, CURLOPT_USERPWD, it->second.c_str());
316 
317  if ((it = p_idx->find("CURLOPT_COOKIEFILE")) != p_idx->end())
318  curl_easy_setopt(curl, CURLOPT_COOKIEFILE, it->second.c_str());
319 
320  if ((it = p_idx->find("CURLOPT_COOKIEJAR")) != p_idx->end())
321  curl_easy_setopt(curl, CURLOPT_COOKIEJAR, it->second.c_str());
322  }
323  c_ret = curl_easy_perform(curl);
324 
325  uint64_t response_code;
326 
327  if (c_ret == CURLE_OK)
328  curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &response_code);
329 
330  curl_easy_cleanup(curl);
331 
332  switch (c_ret) {
333  case CURLE_OK:
334  break;
335  case CURLE_REMOTE_ACCESS_DENIED:
336  case CURLE_AUTH_ERROR:
337  return SERVICE_ERROR_READ_FORBIDDEN;
338  case CURLE_REMOTE_FILE_NOT_FOUND:
339  return SERVICE_ERROR_BLOCK_NOT_FOUND;
340  default:
341  return SERVICE_ERROR_IO_ERROR;
342  }
343 
344  switch (response_code) {
345  case MHD_HTTP_OK:
346  case MHD_HTTP_CREATED:
347  case MHD_HTTP_ACCEPTED:
348  break;
349  case MHD_HTTP_NOT_FOUND:
350  case MHD_HTTP_GONE:
351  return SERVICE_ERROR_BLOCK_NOT_FOUND;
352  case MHD_HTTP_BAD_REQUEST:
353  return SERVICE_ERROR_WRONG_ARGUMENTS;
354  case MHD_HTTP_UNAUTHORIZED:
355  case MHD_HTTP_PAYMENT_REQUIRED:
356  case MHD_HTTP_FORBIDDEN:
357  case MHD_HTTP_METHOD_NOT_ALLOWED:
358  case MHD_HTTP_NOT_ACCEPTABLE:
359  case MHD_HTTP_PROXY_AUTHENTICATION_REQUIRED:
360  case MHD_HTTP_TOO_MANY_REQUESTS:
361  return SERVICE_ERROR_READ_FORBIDDEN;
362  case MHD_HTTP_INTERNAL_SERVER_ERROR ... MHD_HTTP_LOOP_DETECTED:
363  return SERVICE_ERROR_MISC_SERVER;
364  default:
365  return SERVICE_ERROR_IO_ERROR;
366  }
367  if (buff.size() > MAX_BLOCK_SIZE)
368  return SERVICE_ERROR_BLOCK_TOO_BIG;
369 
370  int size = (int) buff.size();
371  int dim[MAX_TENSOR_RANK] = {size, 0};
372  int ret = new_block(p_txn, CELL_TYPE_BYTE, dim, FILL_NEW_DONT_FILL);
373 
374  if (ret == SERVICE_NO_ERROR)
375  memcpy(&p_txn->p_block->tensor.cell_byte[0], buff.data(), size);
376 
377  return ret;
378  }

391  inline StatusCode curl_put(void *url, pBlock p_blk, int mode = WRITE_TENSOR_DATA, Index *p_idx = nullptr) {
392  CURL *curl;
393  CURLcode c_ret;
394 
395  curl = curl_easy_init();
396  if (!curl)
397  return SERVICE_ERROR_NOT_READY;
398 
399  PutBuffer put_buff;
400 
401  switch (mode & (WRITE_TENSOR_DATA | WRITE_C_STR | WRITE_EVERYTHING)) {
402  case WRITE_TENSOR_DATA:
403  put_buff.to_send = p_blk->size*(p_blk->cell_type & 0xff);
404  put_buff.p_base = &p_blk->tensor.cell_byte[0];
405  break;
406 
407  case WRITE_C_STR: {
408  int size = p_blk->size*(p_blk->cell_type & 0xff);
409  put_buff.p_base = &p_blk->tensor.cell_byte[0];
410  put_buff.to_send = strnlen((const char *) put_buff.p_base, size); }
411  break;
412

573  inline bool compose_url(pChar p_dest, pChar p_node, pChar p_url, int buff_size) {
574  int nix;
575 
576  for (MapIS::iterator it = jazz_node_name.begin(); true; ++it) {
577  if (it == jazz_node_name.end())
578  return false;
579 
580  if (it->second == p_node) {
581  nix = it->first;
582  break;
583  }
584  }
585  int ret = snprintf(p_dest, buff_size, "http://%s:%i%s", jazz_node_ip[nix].c_str(), jazz_node_port[nix], p_url);
586 
587  return ret > 0 && ret < buff_size;
588  }
